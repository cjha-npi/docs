//
// TL;DR: In some edge cases, a custom hash can make the default
// side nav disappear.
//
// Doxygen uses NAVTREEINDEX to resolve a hash/href. NAVTREEINDEX is
// split across multiple files and stores entries as: root + hash
// (where "root" is the page URL ending with ".html").
//
// For a given hash/href, Doxygen looks up the root + hash combo in
// NAVTREEINDEX. If it cannot find that combo, it falls back to selecting
// just the root. This works fine with Doxygen's own hashes, but with a
// custom hash it can sometimes fail to find the root as well: while
// searching, it may advance to another NAVTREEINDEX file (because the
// index is split) that does not contain the root entry. In that case,
// Doxygen falls back to index.html, i.e. the "Main Project" item.
//
// This fallback is a serious problem for us because unlike the default
// sidenav, the sidenav generated by doxy-plus in this file does not have
// index.html as a single top level item. So, the fallback to index.html
// can give an incorrect sense of which page we currently are.
//
// ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
//
// We are using the store.js file `store.everything.min.js`. It lets us store
// an expiry time with each key. The expiry is checked only when the key is
// accessed. If the current time is past the expiry time, store.js removes the
// stored data and returns `undefined`.
//
// To set the expiry time, we pass a third parameter to the `set()` function
// provided by store.js. This third parameter is the current time plus the time
// to live in milliseconds. For example:
// `store.set(key, val, Date.now() + 1000)`
//
// This stores `val` under `key` for 1000 milliseconds from now. The key is not
// removed immediately when it expires. It is removed only the next time it is
// accessed after the expiry time has passed.
//
// How the data is stored:
// In localStorage, store.js stores the same logical key using two separate
// keys when using a namespace with expiry support. For example, a key named
// `html_pages` may be stored as:
// `__storejs_F-Doxy-Main_html_pages`
// `__storejs___storejs_F-Doxy-Main_expire_mixin_html_pages`
//
// Using this knowledge, we can iterate through all keys in localStorage and
// extract the namespace name and key name from the second key above. The first
// key format does not give us a clear separator between the namespace name and
// the key name, but the second key does. Its format is:
// `__storejs___storejs_<namespace name>_expire_mixin_<key name>`
//
// So, when iterating through localStorage, we only inspect keys that start
// with `__storejs___storejs_` and then check whether they contain the
// separator `_expire_mixin_`. If both conditions are true, we can extract the
// namespace name and key name.
//
// This lets us find all keys related to any project, as long as they are under
// the same domain, or all local files if no URL is involved. If we want to
// remove an expired key, we simply access it through its namespace name and
// key name. If it has expired, store.js removes it automatically.
//
// ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// of date-time for when the Doxygen output was generated. This date-time
// was inserted by me at the very end of footer.html like this:
//
// <!-- Storing Generation DateTime for use in doxy-plus.js -->
// <script> window.DOXY_PLUS_DATE_TIME = '$datetime'; </script>
//
// <!-- Doxy-Plus Custom JavaScript File -->
// <script type="text/javascript" src="$relpath^doxy-plus.js"></script>
//
// </body>
// </html>
//
// This provided me with custom text like `Thu Apr 16 2026 16:35:06` in this
// (doxy-plus.js) file and I compared it with stored value to check if it was
// a new generation of data. However, the problem with this issue was that
// the window.DOXY_PLUS_DATE_TIME was computed for each html page as it was
// generated by Doxygen, so one page would get `Thu Apr 16 2026 16:35:05` and
// the other would get `Thu Apr 16 2026 16:35:06` because it was generated one
// second later. This created issues because the two texts did not match, resulting
// in my script discarding stored data and regenerating it for no reason.
//
// The solution:
// I found out that even though different html pages had different values for
// window.DOXY_PLUS_DATE_TIME, the footer always shows the exact same value on
// each page. This turns out to be coming from doxygen.css file. In doxygen.css
// file at the very end there is a block of code:
//
// html {
// --timestamp: 'Thu Apr 16 2026 16:57:38'
// }
// span.timestamp { content: ' '; }
// span.timestamp:before { content: var(--timestamp); }
//
// Here the 'span.timestamp:before' is what is shown on the footer as the date-time.
// The "--timestamp: 'Thu Apr 16 2026 16:57:38'" is assigned during the Doxygen
// generation and since all html pages share the same doxygen.css file, all of it
// gets the same date-time value.
//
// So, instead of using my window.DOXY_PLUS_DATE_TIME as the the source to know
// wether the generation is a newly build Doxygen output or is an old one, I can
// use 'getComputedStyle(document.documentElement).getPropertyValue('--timestamp')'
// which gives me the above doxygen.css '--timestamp' value (which is a string value).

// Patch jQuery to not throw when "#id" fragments contain CSS-invalid characters such as '/'.
(function () {
// Purpose
// Patch jQuery so that simple "#id" selectors created from Doxygen hash fragments do not throw.
// Doxygen (notably navtree.js) sometimes does $(hash) where hash comes from a link fragment.
// With some settings (e.g. INLINE_SIMPLE_STRUCTS=YES in doxyfile), those fragments can contain
// '/' and other characters that are not valid in a raw CSS id selector. jQuery/Sizzle then throws:
// "Syntax error, unrecognized expression: #d5/d42/..."
// The exception aborts the click handler and can break navigation/highlighting/page-nav logic.
//
// What this patch changes
// It monkey-patches jQuery's internal constructor $.fn.init (used by $()/jQuery()).
// For selectors that are a very simple "#something" (not complex CSS expressions), it escapes
// the id portion so that characters like '/' become '\/'. This makes the selector valid CSS,
// so jQuery returns an empty set instead of throwing.
//
// Scope and safety
// The patch is intentionally narrow:
// - Only touches selector strings that look like a simple "#id" (no spaces, combinators,
// attribute selectors, pseudo selectors, selector lists, etc.).
// - Does not change complex selectors to avoid altering legitimate query semantics.
// - Uses jQuery.escapeSelector when available (jQuery 3+), otherwise uses a conservative
// fallback escaping rule.
//
// Load order
// This code installs immediately if jQuery is already present. If not, it retries briefly
// using setInterval until jQuery loads, then patches once and stops retrying.
//
// Behavioral impact
// - Eliminates the "unrecognized expression" console errors for invalid hash-based selectors.
// - Prevents exceptions inside Doxygen handlers, allowing navigation logic to continue.
// - If the escaped id does not exist in the DOM, jQuery returns an empty set (normal behavior).
function escapeIdSelector(id) {
// Prefer jQuery's official escaper when present (jQuery 3+).
if (window.jQuery && typeof window.jQuery.escapeSelector === "function") {
return window.jQuery.escapeSelector(id);
}
// Fallback: escape anything that is not [A-Za-z0-9_-]
return String(id).replace(/[^a-zA-Z0-9_-]/g, function (ch) { return "\\" + ch; });
}
function patchOnce() {
var $ = window.jQuery;
if (!$ || !$.fn || !$.fn.init) return false;
if ($.fn.init.__doxyPlusPatched) return true;
var origInit = $.fn.init;
function isSimpleIdSelector(s) {
// Only rewrite very simple "#something" selectors.
// Do NOT touch complex selectors (spaces, combinators, attrs, pseudos, commas).
return typeof s === "string" &&
s.length >= 2 &&
s.charAt(0) === "#" &&
s.indexOf(" ") === -1 &&
s.indexOf(">") === -1 &&
s.indexOf("+") === -1 &&
s.indexOf("~") === -1 &&
s.indexOf(",") === -1 &&
s.indexOf("[") === -1 &&
s.indexOf(":") === -1;
}
var patchedInit = function (selector, context, root) {
if (isSimpleIdSelector(selector)) {
var rawId = selector.slice(1);
selector = "#" + escapeIdSelector(rawId);
}
return origInit.call(this, selector, context, root);
};
patchedInit.prototype = origInit.prototype;
patchedInit.__doxyPlusPatched = true;
$.fn.init = patchedInit;
return true;
}
// Install ASAP, even if our file loads before jquery.js.
if (patchOnce()) return;
var t = setInterval(function () { if (patchOnce()) clearInterval(t); }, 10);
})();

const STORE_JS_NORMAL_PREFIX = '__storejs_'; // Prefix for value keyname in local storage

const PROJ_NAMESPACE = (() => {
// Create a stable hash from the original CONSTS.DOC_ROOT.
// - The same CONSTS.DOC_ROOT always produces the same hash.
// - The hash is always 8 hexadecimal characters.
let hashSeed = 2166136261;
for (let ii = 0; ii < CONSTS.DOC_ROOT.length; ++ii) {
hashSeed ^= CONSTS.DOC_ROOT.charCodeAt(ii);
hashSeed = Math.imul(hashSeed, 16777619);
}
const hash = (hashSeed >>> 0).toString(16).padStart(8, '0');

.replace(/-+/g, '-')
.replace(/^-+|-+$/g, '');
// Combine both parts to create the project namespace.
// Fall back to "proj" if sanitization removes the entire base.
return `${base || 'proj'}--${hash}`;
})();

// Returns key list divided by namespace on this origin (only keys with expiry are included)
function getAllNamespaceKeyPairs() {
const map = new Map();
for (const fullKey of Object.keys(localStorage)) {
if (!fullKey.startsWith(STORE_JS_EXPIRY_PREFIX)) continue;
const sepPos = fullKey.indexOf(STORE_JS_EXPIRY_SEPARATOR, STORE_JS_EXPIRY_PREFIX.length);
if (sepPos === -1) continue;
const namespace = fullKey.slice(STORE_JS_EXPIRY_PREFIX.length, sepPos);
const actualKey = fullKey.slice(sepPos + STORE_JS_EXPIRY_SEPARATOR.length);
if (!namespace || !actualKey) continue;
if (!map.has(namespace)) map.set(namespace, []);
map.get(namespace).push(actualKey);
}
const list = [];
for (const [namespace, keyList] of map) {
list.push({ proj: namespace, keys: keyList });
}
return list;
}
// Returns all keys saved with namespace and expiry at this origin
function getOriginStorageTable() {
const list = getAllNamespaceKeyPairs();
list.sort((a, b) => a.proj.localeCompare(b.proj));
for (const node of list) {
node.keys.sort((a, b) => a.localeCompare(b));
}
const rows = [];
for (const node of list) {
const projName = node.proj;
const projStore = store.namespace(projName);
for (const name of node.keys) {

rows.push({
'Namespace': projName,
'Key': name,
'Value': projStore.get(name),
'Expiry': expiresAt != null ? new Date(expiresAt).toLocaleString() : null
});
}
}
return rows;
}

// Purge any key + data which is stored on this origin and has been expired
function purgeExpired() {
// Use the UTC calendar date in YYYY-MM-DD form so the once-per-day
// guard is compact and timezone-stable.
const today = new Date().toISOString().slice(0, 10);
// Skip the purge if it already completed today.
const lastPurged = localStorage.getItem(KEY__EXPIRED_DATA_LAST_PURGE_DATE);
if (lastPurged === today) return;
// Collect store.js namespace/key pairs for entries that use the
// expire mixin encoding in localStorage.
const list = getAllNamespaceKeyPairs();
// Remove expired data
const now = Date.now();
for (const node of list) {
const nsTemp = store.namespace(node.proj);
for (const keyName of node.keys) {
const exp = nsTemp.getExpiration(keyName);
if (typeof exp === 'number' && exp <= now) {
nsTemp.remove(keyName);
}
}
}
// Record that today's purge completed.
localStorage.setItem(KEY__EXPIRED_DATA_LAST_PURGE_DATE, today);
}

// Deep-clone a nested array structure that contains only arrays and primitive values.
function deepCloneArray(item) {
if (!Array.isArray(item)) return item;
const out = [];
for (const node of item) out.push(deepCloneArray(node));
return out;
}

// Compares two strings and provides highest priority to alphabets.
function compareAlphaBeforePunctuation(a, b) {
// Provided for sorting anonymous namespace after normal normal ones.
// Anonymous namespace is often replaced using 'prettierAnon()' function
// above. Then the sorting happens and so normal namespace that starts
// with alphabets will be sorted above anonymous namespaces. For example
// 'some_ns' will be above '<anon_ns>'.
if (typeof a !== 'string' || typeof b !== 'string') return 0;
const aLen = a.length;
const bLen = b.length;
const minLen = Math.min(aLen, bLen);
// Compare each character till the length of the shorter string
for (let ii = 0; ii < minLen; ++ii) {
const aCh = a[ii].toUpperCase();
const bCh = b[ii].toUpperCase();
const aIsAlpha = (aCh >= 'A' && aCh <= 'Z');
const bIsAlpha = (bCh >= 'A' && bCh <= 'Z');
if (aIsAlpha && !bIsAlpha) return -1;
if (!aIsAlpha && bIsAlpha) return 1;
// If both characters are alphabets then use native compare
const cmpResult = aCh.localeCompare(bCh, undefined, { sensitivity: 'base' });
if (cmpResult !== 0) return cmpResult;
}
// In case of one string is a sub-string of another, e.g. a = 'Some' and
// b = 'SomeFunc', then sort shorter string before longer string.
return aLen - bLen;
}
// Recursively freezes the given value and everything nested inside it.
function deepFreeze(value) {
// Why this is needed:
// Object.freeze(...) is shallow. It only freezes the current object/array itself,
// not the nested objects or arrays stored inside it. So if an object contains
// another object or array, those inner values would still remain mutable unless
// they are also frozen explicitly.
//
// What this function does:
// 1. Checks whether the passed value is a non-null object/array.
// 2. Skips values that are primitives (string, number, boolean, null, etc.).
// 3. Skips values that are already frozen to avoid useless repeated work.
// 4. Freezes the current object/array itself.
// 5. Recursively walks through all nested children and freezes them too.
//
// For arrays:
// It iterates through each item and recursively freezes that item.
//
// For plain objects:
// It gets all own property names using Object.keys(...) and recursively freezes
// the value stored in each property.
//
// Result:
// After this function completes, the full reachable object/array tree is frozen,
// not just the top-level container.
//
// Important:
// - This only works for values reachable from the passed root.
// - This is runtime protection, not compile-time enforcement.
// - In strict mode, illegal mutations throw errors. Without strict mode, some
// mutation attempts may fail silently.
if (value && typeof value === 'object' && !Object.isFrozen(value)) {
// Freeze the current object/array first.
Object.freeze(value);
if (Array.isArray(value)) {
// For arrays, recurse into each element.
for (const item of value) {
deepFreeze(item);
}
} else {
// For objects, Object.keys(value) returns an array of this object's own
// property names, e.g. ['label', 'scope', 'value'].
for (const key of Object.keys(value)) {
// value[key] gets the actual value stored in that property, and that
// nested value must also be frozen if it is an object/array.
deepFreeze(value[key]);
}
}
}
return value;
}
// Waits for the first DOM element that matches selector.
function waitFor(selector) {
// Returns a Promise that:
// 1. resolves with the matching element as soon as it is found,
// 2. resolves with null if the timeout is reached first,
// 3. rejects if selector is not a valid querySelector() string.
//
// Behavior:
// 1. Check immediately in case the element already exists.
// 2. If not found, watch DOM changes and check again after each mutation.
// 3. Stop watching as soon as the element is found or the timeout expires.
return new Promise((resolve, reject) => {
let timer;
let obs;
// Function for cleaning up
function cleanup() {
if (obs) obs.disconnect();
clearTimeout(timer);
}
// Function to find the selector
function findNow() {
try {
const el = document.querySelector(selector);
if (!el) return null;
cleanup();
//console.log(`waitFor(): Found selector "${selector}"`, el);
resolve(el);
return el;
} catch (err) {
cleanup();
console.error(`waitFor(): invalid selector "${selector}"`, err);
reject(err);
return null;
}
}
// Check once before starting the observer.
if (findNow()) return;
// Create a mutation observer that calls findNow on mutations
obs = new MutationObserver(() => { findNow(); });
// Observe a root node that exists even if body is not available yet.
obs.observe(document.documentElement || document, { childList: true, subtree: true });
// Stop waiting after the timeout.
timer = setTimeout(() => {
cleanup();
console.warn(`waitFor(): Timed out waiting for selector "${selector}" after ${CONSTS.VAL.TIMEOUT_MS} ms`);
resolve(null);
}, CONSTS.VAL.TIMEOUT_MS);
});
}
// Check correct construction of html/hash data and decode escaped page titles.
function checkData(kind = 'html') {
const isHtml = kind === 'html';
const isHash = kind === 'hash';
if (!isHtml && !isHash) dispFatalError(`checkData(): Invalid kind "${kind}".`);
const data = isHtml ? vars.htmlData : vars.hashData;
const dataName = isHtml ? 'HTML Data' : 'Hash Data';
const dataVarName = isHtml ? 'vars.htmlData' : 'vars.hashData';
const nodeLen = isHtml ? 3 : 2;
const valIdx = isHtml ? 2 : 1;
const finalEntry = isHtml ? 'Third' : 'Second';
if (!Array.isArray(data)) {
dispFatalError(`checkData(): Top-level ${dataName} (${dataVarName}) is missing or invalid.`);
}
if (data.length === 0) {
dispFatalError(`checkData(): Top-level ${dataName} (${dataVarName}) is empty.`);
}
const decodeText = document.createElement('textarea');
function checkBranch(branch, parentChain = '') {
let idx = 0;
for (const node of branch) {
idx += 1;
const rawPrefix = parentChain
? `checkData(): ${dataName} Node [${dataVarName} > ${parentChain} > Node ${idx}] `
: `checkData(): ${dataName} Node [${dataVarName} > Top Level Node ${idx}] `;
if (!Array.isArray(node)) {
dispFatalError(`${rawPrefix} It is not an array.`);
}
if (node.length !== nodeLen) {
dispFatalError(`${rawPrefix} Its length is not ${nodeLen}, it is ${node.length}.`);
}
if (typeof node[0] !== 'string') {
dispFatalError(`${rawPrefix} First entry (i.e. name) is not a string, it is ${typeof node[0]}.`);
}
// Doxygen changes certain characters like '&' to '&amp;', '<' to '&lt;', etc.
// To change it back we use the below method.
if (node[0].includes('&')) {
decodeText.innerHTML = node[0];
node[0] = decodeText.value;
}
if (node[0].trim().length === 0) {
dispFatalError(`${rawPrefix} First entry (i.e. name) is an empty or whitespace-only string.`);
}
const labelPrefix = parentChain
? `checkData(): ${dataName} Node [${dataVarName} > ${parentChain} > ${idx}: ${node[0]}] `
: `checkData(): ${dataName} Node [${dataVarName} > ${idx}: ${node[0]}] `;
if (isHtml) {
if (typeof node[1] !== 'string' && node[1] !== null) {
dispFatalError(`${labelPrefix} Second entry (i.e. prefix) is of invalid type, it is ${typeof node[1]}.`);
}
if (typeof node[1] === 'string' && node[1].trim().length === 0) {
dispFatalError(`${labelPrefix} Second entry (i.e. prefix) is an empty or whitespace-only string.`);
}
}
if (typeof node[valIdx] === 'string') {
if (node[valIdx].trim().length === 0) {
dispFatalError(`${labelPrefix} ${finalEntry} entry (i.e. link or kids) is an empty or whitespace-only string.`);
}
}
else if (Array.isArray(node[valIdx])) {
if (node[valIdx].length === 0) {
dispFatalError(`${labelPrefix} ${finalEntry} entry (i.e. link or kids) is an empty child branch.`);
}
const fullNameChain = parentChain
? `${parentChain} > ${idx}: ${node[0]}`
: `${idx}: ${node[0]}`;
checkBranch(node[valIdx], fullNameChain);
}
else {
dispFatalError(`${labelPrefix} ${finalEntry} entry (i.e. link or kids) is of invalid type, it is ${typeof node[valIdx]}.`);
}
}
}
checkBranch(data);
}

// Returns a fully loaded, pruned clone of Doxygen NAVTREE, or [] on timeout.
async function buildPrunedDoxygenNavTree() {
// Steps:
// 1. Wait until window.NAVTREE exists and window.NAVTREE[0][2] is a non-empty array (timeout after CONSTS.VAL.TIMEOUT_MS).
// 2. Recursively clone the tree structure into a new array.
// 3. Dynamically load any external JS files for deferred child nodes.
// 4. Recursively attach those loaded children.
// 5. Prune hash entries from the cloned NAVTREE, e.g. remove links like "file.html#anchor".
// 6. Return the pruned clone.
// Wait for NAVTREE
const start = performance.now();
while (!Array.isArray(window.NAVTREE) || !Array.isArray(window.NAVTREE?.[0]?.[2]) || window.NAVTREE[0][2].length === 0) {
const elapsed = performance.now() - start;
if (elapsed > CONSTS.VAL.TIMEOUT_MS) {
dispFatalError(`buildPrunedDoxygenNavTree(): Timed out waiting for a non-empty "window.NAVTREE[0][2]" after ${Math.round(elapsed)} ms`);
return [];
}
//console.log(`buildPrunedDoxygenNavTree(): Waiting for non-empty "window.NAVTREE[0][2]" after ${Math.round(elapsed)} ms...`);
await new Promise((resolve) => { setTimeout(resolve, 16); });
}
// Load a script only once and share the same promise across callers.
function loadScript(relUrl) {
// - Build one absolute URL so different relative forms map to one key.
// - Return the existing promise when the same script was already requested.
// - Remove the failed entry from the cache so a later retry is possible.
// Build the absolute URL.
const fullUrl = new URL(relUrl, CONSTS.DOC_ROOT).href;
// Reuse the existing promise when this absolute URL was already requested.
const existingPromise = vars.scriptLoadMap.get(fullUrl);
if (existingPromise) return existingPromise;
const promise = new Promise((resolve, reject) => {
// Create and configure a script element.
const s = document.createElement('script');
s.src = fullUrl;
s.async = true;
// when the script finishes loading, call resolve()
s.onload = () => { resolve(); };
// if the script fails to load, delete from map, remove it and call reject(err)
s.onerror = (err) => {
vars.scriptLoadMap.delete(fullUrl);
s.remove();
reject(err);
};
// Insert the script element into the page and start the request.
document.head.appendChild(s);
});
// Add script to the map and return the promise
vars.scriptLoadMap.set(fullUrl, promise);
return promise;
}
// Walks the cloned tree, loads deferred children, and replaces string refs
function loadChildren(branch) {
if (!Array.isArray(branch) || branch.length === 0) return Promise.resolve();
const promises = [];
for (const node of branch) {
if (!Array.isArray(node)) continue;
const h = node[1]; // Get the href
const c = node[2]; // Child reference or child array.
if (typeof c === 'string' && c.length > 0) {
// c is a script base name: load it then replace node[2] with the loaded array
// Check if the h is an anchor link (contains '#').
// Inlined elements link to anchors, not new pages.
const isAnchorLink = h && h.includes('#');
// WARNING: Do not remove this check for anchor link. For inlined element
// (e.g., INLINE_SIMPLE_STRUCTS = YES) there is still a script in node[2]
// but loading that script leads to errors because the file it points to
// does not exist.
if (isAnchorLink) {
// This is an inlined element (e.g., INLINE_SIMPLE_STRUCTS = YES).
// It won't have a separate .js file for children.
// Set children to an empty array and skip loading.
node[2] = [];
} else {
// This is a regular page, load the script as before.
// c is a script base name: load it then replace node[2] with the cloned loaded array
promises.push(
loadScript(c + '.js')
.then(() => {
let arr = window[c];
if (!Array.isArray(arr)) arr = window[c.split('/').pop()];
if (!Array.isArray(arr)) {
console.warn(`buildPrunedDoxygenNavTree() - loadChildren(): "${c}.js" loaded, but no valid array global was found for "${c}"`);
node[2] = [];
return;
}
node[2] = deepCloneArray(arr);
return loadChildren(node[2]);
})
.catch((err) => {
console.warn(`buildPrunedDoxygenNavTree() - loadChildren(): Failed to resolve children for "${c}.js"`, err);
node[2] = [];
})
);
}
}
else if (Array.isArray(c) && c.length > 0) {
// Already an array: recurse
promises.push(loadChildren(c));
}
}
return Promise.all(promises);
}
// Clone the default Doxygen NAVTREE
const clonedTree = deepCloneArray(window.NAVTREE);
// Load all deferred children; not all NAVTREE data is present up front.
await loadChildren(clonedTree);
// Prune cloned tree in place.
(function pruneClonedTree(branch) {
if (!Array.isArray(branch) || branch.length === 0) return;
for (let ii = branch.length - 1; ii >= 0; --ii) {
// Remove nodes that are not arrays
if (!Array.isArray(branch[ii])) {
branch.splice(ii, 1);
continue;
}
// Normalize length to 3 entries only
branch[ii].length = 3;
// Remove nodes with empty names
if (typeof branch[ii][0] !== 'string' || branch[ii][0].length === 0) {
branch.splice(ii, 1);
continue;
}
// Prune child nodes if present.
pruneClonedTree(branch[ii][2]);
if (!Array.isArray(branch[ii][2]) || branch[ii][2].length === 0) branch[ii][2] = null;
// Validate href: keep only valid html links, otherwise set to null
if (typeof branch[ii][1] !== 'string' || branch[ii][1].includes('#') || !/\.html$/i.test(branch[ii][1])) branch[ii][1] = null;
// Remove nodes that have no href and no kids
if (branch[ii][1] === null && branch[ii][2] === null) branch.splice(ii, 1);
}
})(clonedTree);
//console.log('buildPrunedDoxygenNavTree(): SUCCESS');
return clonedTree;
}

// Initialize vars.htmlData array
async function initHtmlData() {
// Reset the HTML data before rebuilding it, while maintaining its reference
vars.htmlData.length = 0;
// Check whether the page should be redirected.
function checkRedirect(tree) {
// tree: Array with nodes in the shape [label, scope, hrefOrKids].
// The href stored here is the default NAVTREE href generated by Doxygen,
// for example: d2/d0d/classnpi_1_1_ref_ptr.html.
// We match only the base name, for example: classnpi_1_1_ref_ptr.
// Gate 1: Only proceed on a truly fresh tab (not after reload/back/forward)
const isStale = (sessionStorage.getItem('is_stale') === 'true');
sessionStorage.setItem('is_stale', 'true');
if (isStale) return false; // Not a fresh load: no redirect
// Gate 2: Only on standard navigation (ignore reloads and history traversals)
const navType = performance.getEntriesByType('navigation')[0]?.type ?? 'navigate';
if (navType !== 'navigate') return false; // Reload or back/forward: no redirect
// Gate 3: Only redirect on landing page (root or 'index.html')
const pName = window.location.pathname;
if (!pName.endsWith('/') && !pName.endsWith('/index.html')) return false; // Not on landing page: no redirect
// Gate 4: Stored HTML must exist, be non-empty, and not be the current HTML name
const htmlPrevName = storage.htmlPrevName.load();
if (!htmlPrevName
|| typeof htmlPrevName !== 'string'
|| htmlPrevName.length === 0
|| htmlPrevName === CONSTS.HTML_NAME) return false; // No redirect
// Find a matching HTML page name.
function findLastHtmlPage(branch) {
if (!Array.isArray(branch) || branch.length === 0) return false;
for (const node of branch) {
if (typeof node[2] === 'string') {
const name = node[2].split('/').pop().replace(/\.[^.]*$/, '');
if (name === htmlPrevName) {
window.location.replace(CONSTS.DOC_ROOT + node[2]); // Redirect
return true; // Stop after the first match
}
}
else if (Array.isArray(node[2])) {
if (findLastHtmlPage(node[2])) return true; // Propagate success upward
}
}
return false;
}
// Return whether a matching page was found.
return findLastHtmlPage(tree);
}
// Load previous HTML data if available
const prevHtmlData = storage.htmlData.load();
// Check for previous HTML data being a non-empty array
if (Array.isArray(prevHtmlData) && prevHtmlData.length > 0) {
// Check for redirection, if true return because the url will be redirected
if (checkRedirect(prevHtmlData)) return;
// Add dev hub class in document body to detect it in CSS file
if (prevHtmlData[prevHtmlData.length - 1][0] === 'Dev Hub') {
document.body.classList.add(CONSTS.CLS.DEV_HUB_AVAILABLE);
}
// Extend expiry timeout of ignored not found HTML arrays
storage.htmlIgnored.extendExpiry();
storage.htmlNotFound.extendExpiry();
// Save current HTML name for future use
storage.htmlPrevName.save(CONSTS.HTML_NAME);

// Returns true if 'need' exactly matches the trailing part of 'have'.
// In other words, 'need' must appear as a contiguous suffix of 'have'.
// Examples:
// - have = ['Z', 'A', 'B', 'C'], need = ['A', 'B', 'C'] -> true
// - have = ['A', 'B', 'C'], need = ['A', 'B', 'C'] -> true
// - have = ['Z', 'Y', 'A', 'B', 'C'], need = ['A', 'B', 'C'] -> true
// - have = ['Z', 'A', 'M', 'B', 'C'], need = ['A', 'B', 'C'] -> false
// Here, 'have' is the currently available path and 'need' is the required
// path suffix stored in the pending Map.
function matchPath(have, need) {
if (need.length === 0) return true;
if (have.length < need.length) return false;
const beg = have.length - need.length;
for (let ii = 0; ii < need.length; ++ii) {
if (have[ii + beg] !== need[ii]) return false;
}
return true;
}
// FIFO work queue used to traverse prunedNavTree and search for pending items.
const queue = [];
// Adds a prunedNavTree node to the queue.
// 'path' contains the names of the node's ancestors.
// The queue stores only the last href segment. For example, if a node href is
// "A/B/c.html", the stored value is "c.html". This allows matching by HTML file
// name only. Doxygen-generated HTML names already encode the ancestor context
// (such as namespace and class), so matching the full path is unnecessary and
// would only reflect the folder structure.
function addToQueue(node, path = []) {
const href = node[1] ? node[1].split('/').pop() : null;
queue.push({ Node: node, Link: href, Path: path.concat(node[0]) });
}
// Add the top-level nodes of prunedNavTree to the queue.

// Current queue index acting as the FIFO head.
let head = 0;
// While the queue still has items and the pending Map is not empty,
// try to match queued nodes against pending entries. Each node is expected
// to satisfy at most one pending entry, so stop after the first match.
while (head < queue.length && pending.size > 0) {
const item = queue[head++];
for (const [key, val] of pending) {
if (matchPath(item.Path, val.Path) && (val.Link === null || val.Link === item.Link)) {
obj[key] = item.Node;
pending.delete(key);
break;
}
}

// then it has to be a non-empty branch.
if (item.Node[2]) {
for (const node of item.Node[2]) addToQueue(node, item.Path);
}
}
return obj;
})();

// Build and return a NEW branch for vars.htmlData from a branch of prunedNavTree.
function convertBranch(branch, parentChain, ...regexList) {
// Output node format:
// - [label, null, kids] for branches
// - [label, null, href] for leaves
// Rules:
// 1. Traverse the input branch recursively.
// 2. Keep a branch only if at least one converted child remains.
// 3. Keep each leaf href at most once per call.
// 4. If regex filters are given, keep only leaves whose href base name
// (file name without path or extension) matches at least one regex.
// 5. If a node has both href and kept children, keep it only as a branch
// and record its href in ignoredList once.
// The parentChain is used only to build readable paths such as
// "A > B > C" for ignored-link reporting.
// Invalid or empty input branch produces an empty result branch.
if (!Array.isArray(branch) || branch.length === 0) return [];
// All extra arguments must be RegExp objects.
// On failure, report a fatal error and return an empty branch.
for (const regex of regexList) {
if (!(regex instanceof RegExp)) {
dispFatalError('initHtmlData() - convertBranch(): Invalid Regex found in regexList parameter', regex);
return [];
}
}
// Returns true if the href base name matches any regex in regexList.
// Example: "a/b/class_foo.html" -> "class_foo".
function testHrefBase(fullHref) {
const baseHref = fullHref.split('/').pop();
const dotPos = baseHref.lastIndexOf('.');
const nameHref = (dotPos >= 0) ? baseHref.slice(0, dotPos) : baseHref;
for (const regex of regexList) {
// Reset lastIndex before each test for /g and /y regex safety.
regex.lastIndex = 0;
if (regex.test(nameHref)) return true;
}
return false;
}
// Tracks hrefs already emitted into the returned branch.
// This prevents duplicate leaf entries within this call.
const seen = new Set();
// Recursively convert one sub-branch into the vars.htmlData format.
// The curParChain is only used for ignored-link reporting.
function convert(subBranch, curParChain) {
if (!Array.isArray(subBranch) || subBranch.length === 0) return [];
const rows = [];
for (const node of subBranch) {
// Human-readable full path for diagnostics / ignored-list reporting.
// curParChain("A > B") + node[0] ("C") => name ("A > B > C")
const name = curParChain ? `${curParChain} > ${node[0]}` : node[0];
// Convert children first to decide whether this node survives as a branch.
const kids = node[2] ? convert(node[2], name) : [];
if (kids.length > 0) {
// Branch case: keep children and ignore the node href in output.
// Record that ignored href once if it exists.
if (node[1] && !seen.has(node[1]) && !seenIgnoredHref.has(node[1])) {
ignoredList.push([name, node[1]]);
seenIgnoredHref.add(node[1]);
}
rows.push([node[0], null, kids]);
}
else if (node[1] && !seen.has(node[1])) {
// Leaf case: keep only unique hrefs.
// If filters exist, the href base name must match one of them.
seen.add(node[1]);
if (regexList.length === 0 || testHrefBase(node[1])) {
rows.push([node[0], null, node[1]]);
}
}
// Otherwise skip the node.
}
return rows;
}
// Convert the input branch and return the new vars.htmlData branch.
return convert(branch, parentChain);
}
// Group a flat converted list into alphabetic buckets and return the NEW list.
function putInAlphabeticBuckets(flatList) {
// Input:
// - flatList is expected to be a converted flat list where each item is a
// node in the form [label, null, hrefOrKids].
// Output:
// - Returns a NEW list of bucket nodes in the form [bucketLabel, null, kids].
// Rules:
// 1. Group items by the uppercased first character of node[0].
// 2. If that character is between 'A' and 'Z', use it as the bucket name.
// 3. All other starting characters go into the 'Other' bucket.
// 4. Buckets are emitted in fixed order: 'A' to 'Z', then 'Other'.
// 5. Empty or invalid input returns an empty list.
// Invalid or empty input produces an empty result.
if (!Array.isArray(flatList) || flatList.length === 0) return [];
// Map each bucket name to the list of nodes that belong to it.
const map = new Map();
for (const node of flatList) {
const firstChar = node[0][0].toUpperCase();
const code = firstChar.charCodeAt(0);
const key = (code >= 65 && code <= 90) ? firstChar : 'Other';
if (!map.has(key)) map.set(key, []);
map.get(key).push(node);
}
// Build the final bucketed output list.
const out = [];
// Emit alphabetic buckets in fixed A-Z order.
for (let cc = 65; cc <= 90; ++cc) {
const key = String.fromCharCode(cc);
const kids = map.get(key);
if (kids && kids.length > 0) {
out.push([key, null, kids]);
}
}
// Emit the non-alphabetic bucket last if it has any items.
const otherKids = map.get('Other');
if (otherKids && otherKids.length > 0) {
out.push(['Other', null, otherKids]);
}
return out;
}

// Formats signatures for C++ types
function formatSignature(text) {
if (typeof text !== 'string') return text;
return text
// Remove space before *, &, &&
.replace(/\s+([*&]{1,2})/g, '$1')
// Ensure space after *, &, &&
.replace(/([*&]{1,2})(?!\s)/g, '$1 ')
// Remove spaces inside <...>
.replace(/<\s+/g, '<')
.replace(/\s+>/g, '>')
.replace(/\s+<\s+/g, '<')
.replace(/\s+>\s+/g, '>')
// Remove space before commas, ensure one after
.replace(/\s+,/g, ',')
.replace(/,(?!\s)/g, ', ')
// Remove space after ( and before )
.replace(/\‍(\s+/g, '(')
.replace(/\s+\‍)/g, ')')
// Remove space before (
.replace(/\s+\‍(/g, '(')
// Add space before and after = in special cases
.replace(/\s*=\s*(default|delete|0)/g, ' = $1')
// Collapse multiple spaces and trim
.replace(/\s{2,}/g, ' ')
// leading "}"
.replace(/^\}+\s*/, '')
// trailing "{"
.replace(/\s*\{+$/, '')
.trim();
}

// this function. In this row the 'abc' part is the hash.
// Sometimes (for templates usually) there is no 'id="r_abc"', instead the class
// name contains 'template' e.g. 'class="memitem:abc template"'. So, in order to
// get the hash we get the 'abc' from 'id="r_abc"' and then use 'row.classList'
// which splits on space and then if a class text from classList starts with
// 'memitem:' we split that and take the rest.
// If we get hash from both classList and id and both hash do not match then we
// log a console warning because this is inconsistent behavior.
// For returning the hash, the hash gotten by 'id="r_abc"' is given priority over
// hash gotten by classList. If neither generates a hash this function returns an
// empty string.
function getMemberHash(row) {
if (!row) return '';
let idHash = '';
let classHash = '';
// Get the hash from id
if (typeof row.id === 'string' && row.id.startsWith('r_') && row.id.length > 2) {
idHash = row.id.slice(2);
}
// Get the hash from classList
for (const cls of row.classList) {
if (cls.startsWith('memitem:') && cls.length > 8) {
classHash = cls.slice(8);
break;
}
}
// If both id and classHash hash is generated then they must match
if (idHash && classHash && idHash !== classHash) {
console.warn('initHashData() - getMemberHash(): row id/class hash mismatch', row, idHash, classHash);
}
// return priority: idHash > classHash.
return idHash || classHash || '';
}
// Collect member signatures and group them into vars.hashData by section header.
// NOT for markdown pages.
// The '<tbody>' of '<table class="memberdecls">' inside '<div class=contents">'
// constains the list of members. This list of members usually looks like
// '<tr class="memitem:abc" id="r_abc">' and this is what is use as 'row' in
// this function.
const tables = Array.from(vars.els.PAGE_CONTENT.querySelectorAll("table.memberdecls"));
tables.forEach((table, idx) => {
const grpSigs = [];
const seenName = new Set();
const seenHref = new Set();
const headerEl = table.querySelector("h2.groupheader");
if (!headerEl) return;
let headName = (headerEl.textContent || '').replace(/\s+/g, ' ').trim();
if (headName.length === 0) headName = `Members ${idx + 1}`;
headName = prettierAnon(headName);
const rows = table.querySelectorAll('tr');
rows.forEach((row) => {
const leftTd =
row.querySelector(':scope > td.memItemLeft, :scope > td.memTemplItemLeft');
const rightTd =
row.querySelector(':scope > td.memItemRight, :scope > td.memTemplItemRight');
if (!leftTd || !rightTd) return;
const memberHash = getMemberHash(row);
if (memberHash.length === 0) return;
const leafHref = `#${memberHash}`;
if (seenHref.has(leafHref)) return;
const lftText = leftTd.innerText.replace(/\s+/g, ' ').trim();
const ritText = rightTd.innerText.replace(/\s+/g, ' ').trim();
const leafNameTemp = `${lftText} ${ritText}`.trim();
let leafName = formatSignature(leafNameTemp);
if (leafName.startsWith('enum')) {
leafName = leafName.replace(/\s*\{[\s\S]*\}/, '').trim();
}
leafName = prettierAnon(leafName);
if (seenName.has(leafName)) return;
seenHref.add(leafHref);
seenName.add(leafName);
grpSigs.push([leafName, leafHref]);
});
// Add this group to hash data if it has entries
if (grpSigs.length > 0) {
vars.hashData.push([headName, grpSigs]);
}
});
// The hash data was generated using member table, this means
// we are on a page with code. Set the correct class lists,
// save the data, freeze vars.hashData and return.
if (vars.hashData.length > 0) {
checkData('hash'); // Checks structure and normallizes names
document.body.classList.add(CONSTS.CLS.HASH_DATA_AVAILABLE);
document.body.classList.add(CONSTS.CLS.HASH_DATA_IS_CODE);
storage.hashData.save(vars.hashData);
storage.hashDataIsCode.save('yes');
deepFreeze(vars.hashData);
return;
}
// Return the numeric heading level for a Doxygen Markdown section heading.
function getDoxSectionLevel(el) {
if (!el || typeof el.tagName !== 'string') return 0;
const m = /^H([1-6])$/i.exec(el.tagName);
return m ? Number(m[1]) : 0;
}
// Return the anchor id for a Doxygen Markdown section heading.
function getDoxSectionAnchorId(el) {
if (!el) return '';
// Normal case:
// <h2 class="doxsection"><a class="anchor" id="..."></a>Title</h2>
const a = el.querySelector(':scope > a.anchor[id], :scope > a[id]');
if (a && a.id) return a.id;
// Fallback: id on the heading itself.
if (el.id) return el.id;
// Fallback: anchor immediately before the heading.
const prev = el.previousElementSibling;
if (prev && prev.matches && prev.matches('a.anchor[id], a[id]') && prev.id) {
return prev.id;
}
return '';
}
// Get on-page heading anchors for Markdown pages.
// NOTE: A parent heading loses its own anchor as soon as it gets a child.
const doxSections = Array.from(vars.els.PAGE_CONTENT.querySelectorAll('.doxsection'));
const doxSeenHref = new Set();
const doxStack = []; // { level: number, node: [name, href|kids] }
for (const sec of doxSections) {
const level = getDoxSectionLevel(sec);
if (level === 0) continue;
const id = getDoxSectionAnchorId(sec);
if (!id) continue;
let name = (sec.textContent || '').replace(/\s+/g, ' ').trim();
if (name.length === 0) continue;
name = prettierAnon(name);
const href = `#${id}`;
if (doxSeenHref.has(href)) continue;
doxSeenHref.add(href);
const node = [name, href];
while (doxStack.length > 0 && doxStack[doxStack.length - 1].level >= level) {
doxStack.pop();
}
if (doxStack.length === 0) {
vars.hashData.push(node);
} else {
const parent = doxStack[doxStack.length - 1].node;
// By design, once a heading gets children, its own hash is discarded.
if (!Array.isArray(parent[1])) parent[1] = [];
parent[1].push(node);
}
doxStack.push({ level, node });
}

// markdown pages. In this case we do not set the class and
// data for 'hashDataIsCode'. Just save the class list and

if (vars.hashData.length > 0) {
checkData('hash'); // Checks structure and normallizes names
document.body.classList.add(CONSTS.CLS.HASH_DATA_AVAILABLE);
storage.hashData.save(vars.hashData);
deepFreeze(vars.hashData);

}

// We store the empty sentinel explicitly for hash data
// so that next reload of this page does not try to regenrate
// the hash data and arrive at the same conclusion.
storage.hashData.save('Empty');
}

// Build <ul>...<li> trees from vars.htmlData and vars.hashData
function buildTrees() {
// Class list
const CLS = {
list: 'dpt-list', // ul
node: 'dpt-node', // li, tree node
row: 'dpt-row', // clickable row (<button> for branch, <a> for leaf)
toggle: 'dpt-toggle', // branch toggle icon
prefix: 'dpt-prefix', // prefix, e.g. spe::
label: 'dpt-label', // visible node text
sep: 'dpt-sep', // Separator i.e. span
leaf: 'dpt-node--leaf', // Node with no children
branch: 'dpt-node--branch', // Node with children
current: 'dpt-node--current', // Current node
visited: 'dpt-node--visited', // Visited node (only for hashTree)
expanded: 'dpt-node--expanded', // Expanded branch node
};
// Map from hash href to its constructed leaf node.
const hashToNode = new Map();
// Handle only expand/collapse clicks; links work automatically.
function clickHandler(evt) {
// Find the target
const target = evt.target instanceof Element ? evt.target : null;
if (!target) return;
// Return unless the clicked row is a branch button.
const row = target.closest(`.${CLS.row}`);
if (!(row instanceof HTMLButtonElement)) return;
// Find the node
const li = row.closest(`.${CLS.node}`);
if (!li || !li.classList.contains(CLS.branch)) return;
// Toggle its expanded state
const isExpanded = li.classList.toggle(CLS.expanded);
row.setAttribute('aria-expanded', isExpanded ? 'true' : 'false');
}
// Mark a node as current and expand its parent chain.
function markAndExpandCurrentNode(curNode) {
if (!curNode) return;
curNode.classList.add(CLS.current);
let curPar = curNode.parentElement ? curNode.parentElement.closest(`.${CLS.node}`) : null;
while (curPar) {
if (curPar.classList.contains(CLS.branch)) {
curPar.classList.add(CLS.expanded);
const row = curPar.firstElementChild;
if (row instanceof HTMLButtonElement) row.setAttribute('aria-expanded', 'true');
}
curPar = curPar.parentElement ? curPar.parentElement.closest(`.${CLS.node}`) : null;
}
}
// Handle on-page hash changes.
function onHashChange() {
// Find previous current node first
const prev = vars.els.DP_LEAF_SHOW.querySelector(`.${CLS.node}.${CLS.current}`);
// Read the current hash
const hash = window.location.hash;
// If hash is empty then clear previous current and stop
if (!hash) {
if (prev) {
prev.classList.remove(CLS.current);
prev.classList.add(CLS.visited);
}
return;
}
// Find node directly from map
const item = hashToNode.get(hash) ?? null;
// Clear previous item only if it is different from the new one
if (prev && prev !== item) {
prev.classList.remove(CLS.current);
prev.classList.add(CLS.visited);
}
// Return if new item is not found
if (!item) return;
// Clear visited state from the new item
item.classList.remove(CLS.visited);
// Mark it current and expand its ancestor path
markAndExpandCurrentNode(item);
// Scroll into view
item.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
}
// Build a tree from the given data and config.
function buildTreeFromData(data, cfg) {
if (!Array.isArray(data) || data.length === 0) return null;
let curNode = null;
// Create root <ul>.
const root = document.createElement('ul');
root.classList.add(CLS.list);
// Stack used for iterative tree construction.
const stack = [{ branch: data, parent: root }];
while (stack.length) {
const { branch, parent } = stack.pop();
if (!Array.isArray(branch) || branch.length === 0) continue;
const fragment = document.createDocumentFragment();
for (let ii = 0; ii < branch.length; ++ii) {
const item = branch[ii];
// li for the node
const li = document.createElement('li');
li.classList.add(CLS.node);
// Obtain prefix text (namespace/scope) and value (href or kids)
const prefixText = cfg.getPrefixText(item);
const value = cfg.getValue(item);
// Create prefix as span if prefix text exists
let prefix = null;
if (prefixText) {
prefix = document.createElement('span');
prefix.classList.add(CLS.prefix);
prefix.textContent = prefixText;
}
// We separate label based on '::'. This way we can
// style '::' differently using css. Using normal
// non-monospace font for `::` makes it too weak and
// not visible and so we use monospace font for `::`
// and normal font for the rest of the label.
const labels = [];
const labelText = cfg.getLabelText(item);
const labelParts = labelText.split('::');
for (const lblPart of labelParts) {
if (labels.length > 0) {
const sep = document.createElement('span');
sep.classList.add(CLS.sep);
sep.textContent = '::';
labels.push(sep);
}
const label = document.createElement('span');
label.classList.add(CLS.label);
label.textContent = lblPart;
labels.push(label);
}
// If value is of type string then this is a leaf node
if (typeof value === 'string') {
// Mark li as leaf
li.classList.add(CLS.leaf);
// Create row as a link
const row = document.createElement('a');
row.classList.add(CLS.row);
row.href = cfg.getHref(item);
row.title = cfg.getTitle(item);
// Append prefix and labels to row
if (prefix) row.append(prefix);
for(const lbl of labels) row.append(lbl);
// Append row to li
li.appendChild(row);
// Run the optional leaf callback.
if (cfg.onLeaf) cfg.onLeaf(row, li);
// Check for the node being the current node
if (cfg.isCurrent(row)) curNode = li;
}
else if (Array.isArray(value) && value.length > 0) {
// Value is non-empty array i.e. a branch
// Mark li as a branch
li.classList.add(CLS.branch);
// Create row as a button for expand/collapse
const row = document.createElement('button');
row.classList.add(CLS.row);
row.setAttribute('type', 'button');
row.setAttribute('aria-expanded', 'false');
row.title = cfg.getTitle(item);
// Add toggle as span to show expand/collapse icon
const toggle = document.createElement('span');
toggle.classList.add(CLS.toggle);
toggle.textContent = "\u2B9E"; // ⮞, rotated to show open state
toggle.setAttribute('aria-hidden', 'true');
// Append toggle, prefix and labels to row
row.append(toggle);
if (prefix) row.append(prefix);
for(const lbl of labels) row.append(lbl);
// Append row to li
li.appendChild(row);
// Create a ul sub-branch
const twig = document.createElement('ul');
twig.classList.add(CLS.list);
li.appendChild(twig);
// Queue the child branch and its parent <ul> for later processing.
stack.push({ branch: value, parent: twig });
}
else {
// Warn if the node is neither a leaf nor a branch. This should never happen.
console.warn(`${cfg.warnName}: The following node is neither a leaf nor a branch.`, item);
continue;
}
// Append the item <li> to its parent list <ul> fragment
fragment.appendChild(li);
}
// Append fragment to parent <ul>
parent.appendChild(fragment);
}
// Mark and expand the current node
markAndExpandCurrentNode(curNode);
// Return built tree
return root;
}
function normalizePageHref(href) {
let out = href.split('#')[0];
if (out.endsWith('/')) out += 'index.html';
return out;
}
const curHref = normalizePageHref(window.location.href);
const curHash = window.location.hash;
// Tree built from vars.htmlData
const htmlTree = buildTreeFromData(vars.htmlData, {
warnName: 'buildTreeFromData(): vars.htmlData',
getLabelText: (item) => item[0],
getPrefixText: (item) => item[1] ? `${item[1]}::` : null,
getValue: (item) => item[2],
getHref: (item) => CONSTS.DOC_ROOT + item[2],
getTitle: (item) => item[1] ? `${item[1]}::${item[0]}` : item[0],
isCurrent: (row) => normalizePageHref(row.href) === curHref,
});
// Tree built from vars.hashData
const hashTree = buildTreeFromData(vars.hashData, {
warnName: 'buildTreeFromData(): vars.hashData',
getLabelText: (item) => item[0],
getPrefixText: () => null,
getValue: (item) => item[1],
getHref: (item) => item[1],
getTitle: (item) => item[0],
isCurrent: (row) => row.getAttribute('href') === curHash,
onLeaf: (row, li) => {
const hash = row.getAttribute('href');
if (hash) hashToNode.set(hash, li);
},
});
// Mount htmlTree
vars.els.DP_ROOT_PANE.replaceChildren();
if (htmlTree) {
vars.els.DP_ROOT_PANE.appendChild(htmlTree);
htmlTree.addEventListener('click', clickHandler);
}
// Mount hashTree
vars.els.DP_LEAF_SHOW.replaceChildren();
if (hashTree) {
vars.els.DP_LEAF_SHOW.appendChild(hashTree);
hashTree.addEventListener('click', clickHandler);
window.addEventListener('hashchange', onHashChange);
}
}

// Initialize the root and leaf panes and wire up their resizing behavior.
function initDualPane() {
// Timer ID for scheduling a save of both pane widths.
let paneWidthSaveTimer = null;
// Update and apply the root pane width.
function setRootPaneWidth(w) {
if (w !== vars.rootPaneWidth) {
vars.rootPaneWidth = w;
document.documentElement.style.setProperty(CONSTS.PROP.ROOT_PANE_WIDTH, `${w}px`);
}
}
// Update and apply the leaf pane width.
function setLeafPaneWidth(w) {
if (w !== vars.leafPaneWidth) {
vars.leafPaneWidth = w;
document.documentElement.style.setProperty(CONSTS.PROP.LEAF_PANE_WIDTH, `${w}px`);
}
}
// Return whether the leaf pane should be shown.
function isLeafPaneActive() {
return vars.hashData.length > 0
&& vars.showLeafPane
&& (!vars.autoHideLists || window.innerWidth > CONSTS.VAL.LEAF_PANE_AUTO_HIDE_THRESHOLD);
}
// Adjust pane widths on window resize in desktop view.
function checkDualPaneLayout() {
// Return if not in desktop view
if (window.innerWidth < CONSTS.VAL.DESKTOP_VIEW_MIN_WIDTH) return;
// Proceed only if the combined width exceeds the maximum allowed width.
const maxAllowed = window.innerWidth - CONSTS.VAL.GUTTER_W;
const combinedWidth = vars.rootPaneWidth + vars.leafPaneWidth;
if (combinedWidth > maxAllowed) {
// Calculate rebalanced widths.
let nRoot = isLeafPaneActive() ?
Math.floor(maxAllowed * (vars.rootPaneWidth / combinedWidth))
: Math.min(vars.rootPaneWidth, maxAllowed - CONSTS.VAL.MIN_W);
let nLeaf = maxAllowed - nRoot;
// Enforce the minimum width for both panes.
if (nRoot < CONSTS.VAL.MIN_W) {
nLeaf = Math.max(CONSTS.VAL.MIN_W, maxAllowed - CONSTS.VAL.MIN_W);
nRoot = CONSTS.VAL.MIN_W;
}
else if (nLeaf < CONSTS.VAL.MIN_W) {
nRoot = Math.max(CONSTS.VAL.MIN_W, maxAllowed - CONSTS.VAL.MIN_W);
nLeaf = CONSTS.VAL.MIN_W;
}
// Apply the rebalanced widths.
setRootPaneWidth(nRoot);
setLeafPaneWidth(nLeaf);
// Schedule saving both pane widths.
clearTimeout(paneWidthSaveTimer);
paneWidthSaveTimer = setTimeout(() => {
storage.rootPaneWidth.save(vars.rootPaneWidth);
storage.leafPaneWidth.save(vars.leafPaneWidth);
paneWidthSaveTimer = null;
}, CONSTS.VAL.PANE_WIDTH_SAVE_DEBOUNCE_MS);
}
}
// Wire drag-resize behavior for either the root or leaf pane.
function wirePaneResizer(isRootResizer) {
// State for the current drag session.
let raf = null;
let startX = 0;
let startW = 0;
let movBeg = 0;
// Starting widths used to detect changes on pointerup.
let prevRootWidth = vars.rootPaneWidth;
let prevLeafWidth = vars.leafPaneWidth;
// Apply the latest dragged width.
function applyWidth() {
const newW = startW + (movBeg - startX);
let maxAllowed = window.innerWidth - CONSTS.VAL.GUTTER_W;
if (isRootResizer) {
// Adjust root pane width
if (isLeafPaneActive()) {
// If leaf pane is actively being shown then we cannot shrink its
// size to let root pane width grow.
maxAllowed -= vars.leafPaneWidth;
const nVal = Math.max(CONSTS.VAL.MIN_W, Math.min(maxAllowed, newW));
setRootPaneWidth(nVal);
}
else {
// If leaf pane is not actively being shown then we can shrink its
// size to let root pane width grow.
maxAllowed -= CONSTS.VAL.MIN_W;
const nVal = Math.max(CONSTS.VAL.MIN_W, Math.min(maxAllowed, newW));
setRootPaneWidth(nVal);
if (nVal + vars.leafPaneWidth > window.innerWidth - CONSTS.VAL.GUTTER_W) {
setLeafPaneWidth(window.innerWidth - CONSTS.VAL.GUTTER_W - nVal);
}
}
}
else {
// Adjust leaf pane width
maxAllowed -= vars.rootPaneWidth;
const nVal = Math.max(CONSTS.VAL.MIN_W, Math.min(maxAllowed, newW));
setLeafPaneWidth(nVal);
}
// Mark that no frame is currently pending.
raf = null;
}
// Handle pointer movement during drag.
function onMove(evt) {
evt.preventDefault();
movBeg = evt.clientX;
if (raf) return; // One queued frame is enough.
raf = requestAnimationFrame(applyWidth);
}
// Finish dragging and apply the last pending width if needed.
function onUp(evt) {
evt.preventDefault();
// Flush the final width if a frame is still pending.
if (raf) {
cancelAnimationFrame(raf);
applyWidth();
}
// Save widths if they changed during this drag session.
if (prevRootWidth !== vars.rootPaneWidth) storage.rootPaneWidth.save(vars.rootPaneWidth);
if (prevLeafWidth !== vars.leafPaneWidth) storage.leafPaneWidth.save(vars.leafPaneWidth);
document.body.style.cursor = '';
document.removeEventListener('pointermove', onMove);
document.removeEventListener('pointerup', onUp);
document.removeEventListener('pointercancel', onUp);
}
const resizer = isRootResizer ? vars.els.DP_ROOT_RESIZER : vars.els.DP_LEAF_RESIZER;
const getW = isRootResizer ? () => vars.rootPaneWidth : () => vars.leafPaneWidth;
// Start dragging when the resizer is pressed.
resizer.addEventListener('pointerdown', (evt) => {
evt.preventDefault();
startX = evt.clientX;
startW = getW();
prevLeafWidth = vars.leafPaneWidth;
prevRootWidth = vars.rootPaneWidth;
document.body.style.cursor = 'ew-resize';
document.addEventListener('pointermove', onMove, { passive: false });
document.addEventListener('pointerup', onUp, { passive: false });
document.addEventListener('pointercancel', onUp, { passive: false });
}, { passive: false });
}
// Apply the current stored widths before any interaction starts.
document.documentElement.style.setProperty(CONSTS.PROP.ROOT_PANE_WIDTH, `${vars.rootPaneWidth}px`);
document.documentElement.style.setProperty(CONSTS.PROP.LEAF_PANE_WIDTH, `${vars.leafPaneWidth}px`);
// Wire both pane resizers.
wirePaneResizer(true);
wirePaneResizer(false);
// Re-check layout whenever the window size changes.
window.addEventListener('resize', checkDualPaneLayout);
// Correct the layout once immediately on setup.
checkDualPaneLayout();
}

// Set up layout observers and a mutation observer for the search results window.
async function setupObservers() {
// DOM elements that we need in this function
const reqEls = [
'TOP',
'TITLE_AREA',
'SIDE_NAV',
'NAV_PATH',
'DOC_CONTENT',
'CONTAINER',
'SEARCH_RESULTS_WIN'
];
// Return the required element keys whose cached elements are missing or disconnected.
// This avoids awaiting waitFor() for elements that are already available.
function getMissingEls() {
const out = [];
for (const keyName of reqEls) {
if (!vars.els[keyName] || !vars.els[keyName].isConnected) {
out.push(keyName);
}
}
return out;
}
// Wait only for the required elements that are still missing.
async function waitMissingEls(keyNames) {
const jobs = [];
for (const keyName of keyNames) {
const selector = CONSTS.SEL[keyName];
if (!selector) {
vars.els[keyName] = null;
continue;
}
jobs.push(
waitFor(selector).then(el => {
vars.els[keyName] = el || null;
})
);
}

}

// Restore a page's scroll position on load and save its scroll position on scroll and page hide.
function restoreAndObserveVScroll() {
// vars.els.DOC_CONTENT has already been verified and waited for (if needed)
// in the setupObservers() function, so no need to do that here, just
// return if it is not there or if it is not connected.
if (!vars.els.DOC_CONTENT || !vars.els.DOC_CONTENT.isConnected) return;
// Timer ID used to debounce scroll-position saves.
let saveScrollTimer = null;
// Retry restoring until vars.els.DOC_CONTENT is tall enough or the retry limit is reached.
function restoreScroll() {
const savedTop = Number(storage.pageScrollVPos.load());
if (!Number.isFinite(savedTop) || savedTop < 0) return;
let tries = 0;
function tryRestore() {
vars.els.DOC_CONTENT.scrollTop = savedTop;
const maxTop = vars.els.DOC_CONTENT.scrollHeight - vars.els.DOC_CONTENT.clientHeight;
if (maxTop >= savedTop || tries >= 50) return;
tries += 1;
requestAnimationFrame(tryRestore);
}
tryRestore();
}
// Save the current vertical scroll position of vars.els.DOC_CONTENT.
function saveScroll() {
const val = vars.els.DOC_CONTENT.scrollTop;
if (val > 0) storage.pageScrollVPos.save(val);
else storage.pageScrollVPos.remove();
}
// Debounce scroll-triggered saves so storage is written only after scrolling settles.
function onScroll() {
window.clearTimeout(saveScrollTimer);
saveScrollTimer = window.setTimeout(() => {
saveScrollTimer = null;
saveScroll();
}, 200);
}
// Clear any pending debounce timer and save the current scroll position.
function saveScrollNow() {
window.clearTimeout(saveScrollTimer);
saveScrollTimer = null;
saveScroll();
}
// Restore initially
restoreScroll();
// Add listeners for saving
vars.els.DOC_CONTENT.addEventListener('scroll', onScroll, { passive: true });
window.addEventListener('pagehide', saveScrollNow);
}
