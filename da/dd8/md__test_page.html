#define WIN32_LEAN_AND_MEAN
#define NOMINMAX
#include <windows.h>
#include <shlobj.h> // SHCreateDirectoryExW
#include <string>
#include <string_view>
#pragma comment(lib, "Shell32.lib")
namespace
{
bool endsWithSlash(const wchar_t* filePath)
{
if(!filePath) return false;
size_t n = ::wcsnlen_s(filePath, 32768);
if(n == 0) return false;
const wchar_t last = filePath[n - 1];
return last == L'\\' || last == L'/';
}
bool ensureParentDirExists(const wchar_t* filePath)
{
if(filePath == nullptr || filePath[0] == L'\0')
return false;
auto isSep = [](wchar_t ch) noexcept { return ch == L'\\' || ch == L'/'; };
auto isAlpha = [](wchar_t ch) noexcept {
return (ch >= L'A' && ch <= L'Z') || (ch >= L'a' && ch <= L'z');
};
const size_t n = ::wcsnlen_s(filePath, 32768);
if(n == 0 || n >= 32768)
return false;
const bool startsWith2Backslashes = (n >= 2 && filePath[0] == L'\\' && filePath[1] == L'\\');
// Reject device/extended namespaces: "\\?\" and "\\.\"
if(startsWith2Backslashes && n >= 4) {
const wchar_t ns = filePath[2];
if((ns == L'?' || ns == L'.') && isSep(filePath[3]))
return false;
}
// Require absolute: "X:\..." or UNC "\\server\share\..."
const bool isDriveAbs = (n >= 3 && isAlpha(filePath[0]) && filePath[1] == L':' && filePath[2] == L'\\');
const bool isUncAbs = startsWith2Backslashes;
if(!isDriveAbs && !isUncAbs)
return false;
const std::wstring_view path(filePath, n);
const size_t pos = path.find_last_of(L"\\/");
if(pos == std::wstring_view::npos || pos == 0)
return false;
const std::wstring_view dirView = path.substr(0, pos);
if(dirView.empty())
return false;
// "X:\file.log" -> dirView == "X:" (drive root: do not create)
if(dirView.size() == 2 && isAlpha(dirView[0]) && dirView[1] == L':')
return isDriveAbs;
// UNC: never attempt to create the share root "\\server\share"
if(isUncAbs) {
// Find end of server name
const size_t s1 = path.find_first_of(L'\\', 2);
if(s1 == std::wstring_view::npos || s1 == 2)
return false;
// Find end of share name
const size_t s2 = path.find_first_of(L'\\', s1 + 1);
if(s2 == std::wstring_view::npos || s2 == s1 + 1)
return false;
// Parent dir exactly equals "\\server\share" => do not create (share must already exist)
if(dirView.size() == s2)
return true;
}
const std::wstring dir(dirView);
const int rc = ::SHCreateDirectoryExW(nullptr, dir.c_str(), nullptr);
if(rc == ERROR_SUCCESS || rc == ERROR_ALREADY_EXISTS)
return true;
// Robust fallback: accept if it already exists as a directory.
const DWORD attr = ::GetFileAttributesW(dir.c_str());
if(attr != INVALID_FILE_ATTRIBUTES && (attr & FILE_ATTRIBUTE_DIRECTORY) != 0)
return true;
return false;
}
HANDLE openAppendFile(const wchar_t* filePath)
{
return ::CreateFileW(
filePath,
FILE_APPEND_DATA, // atomic append per write
FILE_SHARE_READ, // allow others to read; disallow move/delete/write
nullptr,
OPEN_ALWAYS,
FILE_ATTRIBUTE_NORMAL,
nullptr);
}
}
/** \page entry_points Entry points
* \brief Where execution starts.
*
* Console build: \ref wmain
*/
int wmain(int argc, wchar_t* argv[])
{
if(argc < 2 || argv[1] == nullptr || argv[1][0] == L'\0')
return 1; // Missing/invalid primary log file path argument
const wchar_t* primaryLogFileName = argv[1];
const wchar_t* backupLogFileName = (argc >= 3 && argv[2] != nullptr && argv[2][0] != L'\0') ? argv[2] : nullptr;
if(endsWithSlash(primaryLogFileName)) return 2; // Primary log path ends with slash
if(backupLogFileName && endsWithSlash(backupLogFileName)) backupLogFileName = nullptr; // or return error
const HANDLE hStdIn = ::GetStdHandle(STD_INPUT_HANDLE);
if(hStdIn == INVALID_HANDLE_VALUE || hStdIn == nullptr)
return 3; // Missing/invalid STDIN handle (pipe not redirected/inherited)
HANDLE hLogFile = INVALID_HANDLE_VALUE;
constexpr DWORD bufSize = 64u * 1024u; // 64 KiB
char buffer[bufSize]{};
DWORD bRead = 0;
DWORD bWrite = 0;
DWORD bRemain = 0;
DWORD bOffset = 0;
while(true) {
bRead = 0;
const BOOL readSuccess = ::ReadFile(hStdIn, buffer, bufSize, &bRead, nullptr);
if(!readSuccess) {
const DWORD e = ::GetLastError();
if(e == ERROR_BROKEN_PIPE || e == ERROR_HANDLE_EOF)
break; // Normal termination: parent closed/crashed and pipe ended.
if(hLogFile != INVALID_HANDLE_VALUE)
::CloseHandle(hLogFile);
return 4; // Read failed (unexpected error)
}
if(bRead == 0)
break; // EOF (End of File) i.e. Main application closed
if(hLogFile == INVALID_HANDLE_VALUE) {
// Try primary first.
hLogFile = openAppendFile(primaryLogFileName);
if(hLogFile == INVALID_HANDLE_VALUE) {
const DWORD pErr = ::GetLastError();
if(pErr == ERROR_PATH_NOT_FOUND || pErr == ERROR_DIRECTORY) {
if(ensureParentDirExists(primaryLogFileName))
hLogFile = openAppendFile(primaryLogFileName);
}
}
// Fallback to backup if primary failed.
if(hLogFile == INVALID_HANDLE_VALUE && backupLogFileName != nullptr) {
hLogFile = openAppendFile(backupLogFileName);
// If backup failed due to missing dirs, create dirs and retry once.
if(hLogFile == INVALID_HANDLE_VALUE) {
const DWORD bErr = ::GetLastError();
if(bErr == ERROR_PATH_NOT_FOUND || bErr == ERROR_DIRECTORY) {
if(ensureParentDirExists(backupLogFileName))
hLogFile = openAppendFile(backupLogFileName);
}
}
if(hLogFile == INVALID_HANDLE_VALUE)
return 5; // Failed to open/create both primary and backup (even after mkdir)
}
else if(hLogFile == INVALID_HANDLE_VALUE) {
return 6; // Failed to open/create primary (no backup provided)
}
}
bRemain = bRead;
bOffset = 0;
int noProgressTries = 0;
while(bRemain > 0) {
bWrite = 0;
const BOOL writeSuccess = ::WriteFile(hLogFile, buffer + bOffset, bRemain, &bWrite, nullptr);
if(!writeSuccess) {
::CloseHandle(hLogFile);
return 7; // Write failed
}
if(bWrite == 0) {
// No progress. Retry a few times, then give up.
if(++noProgressTries >= 16) {
::CloseHandle(hLogFile);
return 8; // No progress after many tries
}
::SwitchToThread(); // brief yield
continue;
}
// Progress
noProgressTries = 0;
bOffset += bWrite;
bRemain -= bWrite;
}
}
if(hLogFile != INVALID_HANDLE_VALUE) {
::FlushFileBuffers(hLogFile);
::CloseHandle(hLogFile);
}
return 0; // Normal shutdown
}
