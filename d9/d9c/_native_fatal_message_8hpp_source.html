36 /// \brief Windows native fatal error message box.
37 /// \details This function displays a modal Windows-based message box
38 /// for use when `QApplication` might not be available, e.g.,
39 /// during application startup fatal errors.
40 /// \param title The title of the message box, "[AppName] Fatal Error:" is prepended to this.
41 /// \param msg The message to display in the message box.
42 /// \warning This function does not return; it calls `std::abort()` after
43 /// displaying the message box which leads to abnormal program termination.
44 [[noreturn]] inline void nativeFatalMessage(const QString& title, const QString& msg) noexcept
45 {
46 QString fullTitle = QString::fromLatin1(npi::AppMetaData::Name_) + QStringLiteral(" Fatal Error");
47 if(!title.isEmpty())
48 fullTitle.append(QStringLiteral(": ")).append(title);
49 const std::wstring t = fullTitle.toStdWString();
50 const std::wstring m = msg.toStdWString();
51 ::MessageBoxW(nullptr, m.c_str(), t.c_str(), MB_OK | MB_ICONERROR | MB_TOPMOST | MB_SETFOREGROUND);
52 std::abort();
53 }
