1/**
2* \file Font.hpp
3* \brief Thread-safe font utility for application and fixed-width fonts.
4*
5* \details
6* This header defines npi::Font, a small utility that centralizes font handling:
7* - Maintains a global application font and a global fixed-width font.
8* - Provides style flags (bold, italic, underline, overline, strikeout) and capitalization.
9* - Supports absolute and relative point sizes and computes effective sizes.
10* - Caches QFont and its metrics (QFontMetrics, QFontMetricsF) keyed by style+size.
11* - Uses per-family mutexes to guard static state and caches.
12*
13* \note GUI thread requirement:
14* Any interaction with QWidget or QApplication must occur on the GUI thread.
15* This class posts updates using queued connections where needed.
16*
17* \warning Ordering note for SetAppFont_ and SetFixFont_:
18* These methods update internal state and clear caches under a mutex, unlock,
19* and then perform GUI updates (qApp->setFont or posting QWidget::update()).
20* This introduces a brief window where cache state and GUI state may not be
21* perfectly in sync. This is acceptable in practice and avoids holding a lock
22* across GUI calls.
23*/

45 /**
46 * \brief Thread-safe font utility with style, sizing, and caching.
47 *
48 * \details
49 * The Font class exposes:
50 * - Global setters for the base application font and the base fixed-width font.
51 * - A Style struct to describe per-request font attributes.
52 * - Absolute and relative point-size inputs with effective-size calculation.
53 * - Cached Groups of (QFont, QFontMetrics, QFontMetricsF) for fast reuse.
54 * - Static mutexes to protect shared state and caches.
55 *
56 * Typical use:
57 * \code
58 * // Initialize at app startup (GUI thread):
59 * npi::Font::SetAppFont_(qApp->font());
60 *
61 * // On demand, configure style and size:
62 * npi::Font::Style st;
63 * st.bold_ = true;
64 * npi::Font f{st};
65 * f.setRelativeSize_(+1);
66 *
67 * // Fetch QFont or metrics:
68 * QFont font = f.appFont_();
69 * QFontMetrics fm = f.appFontMetrics_();
70 * \endcode
71 */

139 /**
140 * \brief Set the global application font (family, size, style).
141 *
142 * \details
143 * - Clears the application font cache so new Groups reflect the new base font.
144 * - Calls qApp->setFont(font) after releasing the internal mutex.
145 *
146 * \pre Must be invoked when QApplication exists. Should be invoked on the GUI thread.
147 * \param font New application base font.
148 */
149 static void SetAppFont_(const QFont& font)
150 {
151 AppFontMutex_.lock(); // Lock for thread safety.
152 if(font != AppFont_) { // Only update if the new font differs.
153 AppFont_ = font; // Update the global application font.
154 AppFontSize_ = (font.pointSize() < 0 ? 9 : font.pointSize()); // Update the font size.
155 AppGroupCache_.clear(); // Clear the cache to ensure new font groups are created.
156 AppFontMutex_.unlock(); // Unlock before GUI call.
157 qApp->setFont(font); // Apply the new font to the application.
158 }
159 else {
160 AppFontMutex_.unlock(); // Unlock if no change.
161 }
162 }

164 // Sets the given font as fixed width font (including its size)
165 // - Clears existing Fixed width Font Cache
166 // - Iterates over all widgets and updates those with the
167 // property FixedWidthFontWidget
168 /**
169 * \brief Set the global fixed-width font (family, size, style).
170 *
171 * \details
172 * - Clears the fixed-width font cache.
173 * - Queues `QWidget::update()` on widgets marked with `props::FixedWidthFontWidget`.
174 * Lifetime is guarded via QPointer in the queued lambda.
175 *
176 * \pre Should be invoked on the GUI thread.
177 * \param font New fixed-width base font.
178 */
179 static void SetFixFont_(const QFont& font)
180 {
181 FixFontMutex_.lock(); // Lock for thread safety.
182 if(font != FixFont_) { // Only update if the new font differs.
183 FixFont_ = font; // Update the global fixed-width font.
184 FixFontSize_ = (font.pointSize() < 0 ? 9 : font.pointSize()); // Update the font size.
185 FixGroupCache_.clear(); // Clear the fixed-width font cache.
186 FixFontMutex_.unlock(); // Unlock before GUI calls.
187
188 // Iterate over all widgets in the application.
189 for(const auto widgets = qApp->allWidgets(); const auto widget : widgets) {
190 // If the widget uses the fixed-width font, force an update to apply the new font.
191 if(QPointer<QWidget> w{widget}; w && w->property(GuiProps::FixedWidthFontWidget_).toBool()) {
192 QMetaObject::invokeMethod(qApp, [w] {
193 if(w)
194 w->update();
195 }, Qt::QueuedConnection);
196 }
197 }
198 }
199 else {
200 FixFontMutex_.unlock(); // Unlock if no change.
201 }
202 }

259 // Static function to convert a Color enum into a QColor.
260 // The function takes into account whether the background is dark
261 // which is provided through darkBackground boolean parameter.
262 /**
263 * \brief Map Color to QColor, taking background brightness into account.
264 * \param color Named color.
265 * \param darkBackground True if background is dark.
266 * \return A QColor suitable for the background choice.
267 */
268 static QColor Color_(const Color color, const bool darkBackground)
269 {
270 if(color == Color::Red)
271 return darkBackground ? Qt::red : Qt::darkRed;
272 if(color == Color::Green)
273 return darkBackground ? Qt::green : Qt::darkGreen;
274 if(color == Color::Blue)
275 return darkBackground ? Qt::blue : Qt::darkBlue;
276 if(color == Color::Yellow)
277 return darkBackground ? Qt::yellow : Qt::darkYellow;
278 if(color == Color::Cyan)
279 return darkBackground ? Qt::cyan : Qt::darkCyan;
280 if(color == Color::Magenta)
281 return darkBackground ? Qt::magenta : Qt::darkMagenta;
282 return darkBackground ? Qt::white : Qt::black; // For Default and any unhandled case, return white for dark backgrounds and black for light backgrounds.
283 }

538 /**
539 * \brief Compute a hash key from style and size.
540 *
541 * \details
542 * Combines boolean flags, capitalization, and size into a single 32-bit integer.
543 * Bit layout before qHash:
544 * - bits 0..4 : bold, italic, underline, overline, strikeout
545 * - bits 5..12 : capitalization (8 bits)
546 * - bits 13..28: point size (16 bits, truncated, assumes size is within [0, 65535])
547 *
548 * \param style Style flags and capitalization.
549 * \param size Point size used for the key (truncated to 16 bits).
550 * \param seed Optional hash seed.
551 * \return size_t hash suitable as a QHash key.
552 */
553 static size_t Hash_(const Style& style, const int size, size_t seed = 0)
554 {
555 quint32 combined = 0;
556 combined |= (style.bold_ ? 1u : 0u) << 0;
557 combined |= (style.italic_ ? 1u : 0u) << 1;
558 combined |= (style.underline_ ? 1u : 0u) << 2;
559 combined |= (style.overline_ ? 1u : 0u) << 3;
560 combined |= (style.strikeout_ ? 1u : 0u) << 4;
561 combined |= (static_cast<quint32>(style.capitalization_) & 0xFF) << 5;
562 combined |= (static_cast<quint32>(size) & 0xFFFF) << 13;
563 return qHash(combined, seed);
564 }

566 /**
567 * \brief Get or create the cached Group for the application font family.
568 *
569 * \details
570 * Retrieves or creates a cached Group for the application font. The effective
571 * font size is computed based on either the absolute size (if > 0) or the current
572 * application font size plus the relative size adjustment. A hash key is computed
573 * from the style and size, and the cache is checked for an existing entry. If not
574 * found, a new QFont is configured, stored in a Group, and inserted into the cache.
575 *
576 * \param style Style flags and capitalization.
577 * \param absSize Absolute point size (> 0 overrides).
578 * \param relSize Relative point size delta.
579 * \return Reference to cached Group (valid for the lifetime of the cache).
580 *
581 * \note Uses AppFont_ as the base family.
582 */
583 static const Group& GetAppGroup_(const Style& style, const int absSize, const int relSize)
584 {
585 const int size = absSize > 0 ? absSize : (AppFontSize_ + relSize > 0 ? AppFontSize_ + relSize : AppFontSize_); // Compute effective font size.
586 const size_t hash = Hash_(style, size); // Compute the cache key using the style and effective size.
587 auto it = AppGroupCache_.find(hash); // Look for the key in the application font cache.
588 if(it == AppGroupCache_.end()) { // Not found in cache: create a new QFont based on the current AppFont_.
589 QFont font{AppFont_};
590 font.setPointSize(size);
591 font.setBold(style.bold_);
592 font.setItalic(style.italic_);
593 font.setUnderline(style.underline_);
594 font.setOverline(style.overline_);
595 font.setStrikeOut(style.strikeout_);
596 font.setCapitalization(style.capitalization_);
597 AppGroupCache_.insert(hash, Group{font}); // Insert the new Group into the cache.
598 it = AppGroupCache_.find(hash); // Update iterator after insertion and retrieve for newly inserted group.
599 }
600 return it.value(); // Return the cached Group containing the configured QFont and its metrics.
601 }
