30{
31 static const QString path = []() -> QString {
32 // Determine ProgramData base path
33 QString base;
34
35 // Windows API
36 PWSTR p = nullptr;
37 const HRESULT hr = SHGetKnownFolderPath(FOLDERID_ProgramData, KF_FLAG_DEFAULT, nullptr, &p);
38 if(SUCCEEDED(hr) && p) {
39 base = QString::fromWCharArray(p);
40 CoTaskMemFree(p);
41 }
42
43 // Fallback to env
44 if(base.isEmpty()) {
45 if(const QByteArray env = qgetenv("PROGRAMDATA"); !env.isEmpty())
46 base = QString::fromLocal8Bit(env);
47 }
48
49 // Abort if missing
50 if(base.isEmpty()) {
51 npi::win::abortMsgBox(
52 "ABORT: ProgramData Unavailable",
53 "Unable to determine the ProgramData path.");
54 }
55
56 // Ensure ProgramData exists
57 QDir root(base);
58 if(!root.exists()) {
59 npi::win::abortMsgBox(
60 "ABORT: ProgramData Missing",
61 QString("ProgramData path '%1' does not exist.")
62 .arg(QDir::toNativeSeparators(root.absolutePath())));
63 }
64
65 // Ensure [ProgramData]/[OrganizationName] exists
66 const auto orgName = QString{npi::app::orgDirName};
67 const QString target = root.filePath(orgName);
68 if(!root.mkpath(orgName)) {
69 npi::win::abortMsgBox(
70 "ABORT: Create Directory Failed",
71 QString("Unable to create required directory '%1'.")
72 .arg(QDir::toNativeSeparators(target)));
73 }
74
75 // Normalize and return
76 QString out = QFileInfo(target).canonicalFilePath(); // absolute + symlinks/junctions resolved
77 if(out.isEmpty()) out = QDir::cleanPath(QFileInfo(target).absoluteFilePath()); // defensive fallback

79 }();
