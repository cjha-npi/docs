15 /**
16 * \brief Concept to check if two types are related through inheritance.
17 *
18 * \details
19 * Checks whether two types are related through inheritance,
20 * either in a base-to-derived or derived-to-base direction.
21 * This is useful for verifying safe polymorphic usage such as
22 * upcasting or downcasting between types.
23 *
24 * \par Example:
25 * \code{.cpp}
26 * struct A {};
27 * struct B : A {};
28 * struct C {};
29 * PolymorphicallyRelated<A, B> // True
30 * PolymorphicallyRelated<B, A> // True
31 * PolymorphicallyRelated<A, A> // True
32 * PolymorphicallyRelated<A, C> // False
33 * \endcode
34 **/
35 template <typename A, typename B>
36 concept PolymorphicallyRelated =
37 std::is_base_of_v<A, B> || std::is_base_of_v<B, A>;

39 /**
40 * \brief Concept to check if a type supports shared_from_this().
41 *
42 * \details
43 * Checks if a given type T inherits from `std::enable_shared_from_this<T>`
44 * so that `shared_from_this()` can be safely called on an object of type T.
45 *
46 * \par Example:
47 * \code{.cpp}
48 * struct MyStruct : std::enable_shared_from_this<MyStruct> {};
49 * SupportsSharedFromThis<MyStruct> // true
50 * SupportsSharedFromThis<std::string> // false
51 * \endcode
52 **/
53 template <typename T>
54 concept SupportsSharedFromThis =
55 std::derived_from<T, std::enable_shared_from_this<T>>;
