328 {
329 if(hImpLogProc == nullptr ||

334 return;
335 }

338 size_t remaining = len;

340 int noProgressTries = 0;
341
342 while(remaining > 0) {
343 const size_t chunkSz = (remaining > static_cast<size_t>(0xFFFFFFFFu))

685 void appendMsg(char* out, size_t& pos, const QString& msg) noexcept
686 {
687 if(!out || pos >= txtCap - 3) return;
688
689 const ushort* txt = msg.utf16(); // no allocation
690 const qsizetype qlen = msg.size(); // no allocation
691 if(!txt || qlen <= 0) return;
692 const size_t len = static_cast<size_t>(qlen);
693 const size_t nmax = (len < txtCap) ? len : txtCap;
694
695 const size_t pos0 = pos;
696 for(size_t ii = 0; ii < nmax && pos < txtCap - 3; ++ii)
697 appendChr(out, pos, static_cast<wchar_t>(txt[ii]));
698
699 while(pos > pos0 && out[pos - 1] == ' ')
700 --pos;
701
702 out[pos] = '\0';
703 }

792 LONG WINAPI windowsExceptionFilter(EXCEPTION_POINTERS* ep)
793 {
794 if(!ep || !ep->ExceptionRecord)
795 return EXCEPTION_CONTINUE_SEARCH;
796
797 const DWORD code = ep->ExceptionRecord->ExceptionCode;
798
799 if(code == EXCEPTION_BREAKPOINT || code == EXCEPTION_SINGLE_STEP)
800 return EXCEPTION_CONTINUE_SEARCH;
801
802 if(::InterlockedExchange(&crashing, 1) != 0) {
803 ::TerminateProcess(::GetCurrentProcess(), static_cast<UINT>(code));
804 return EXCEPTION_EXECUTE_HANDLER;
805 }
806
807 cTxtPos = 0;
808 cTxtBuf[0] = '\0';
809
810 // CASE 1: Stack Overflow (Critical / Low Stack)
811 if(code == EXCEPTION_STACK_OVERFLOW) {
812 if(_resetstkoflw()) {
813 // Reset guard page to proceed safely (include malloc.h)
814 appendTxt(cTxtBuf, cTxtPos, "Windows (SEH) Exception | Stack Overflow (Recursion or Stack Exhaustion)");
815 impLogWrite(cTxtBuf, cTxtPos, true);
816 }
817 // Kill immediately - we cannot recover safely
818 ::TerminateProcess(::GetCurrentProcess(), static_cast<UINT>(code));
819 return EXCEPTION_EXECUTE_HANDLER;
820 }
821
822 // CASE 2: Standard SEH Crash (Stack is Safe)
823 appendTxt(cTxtBuf, cTxtPos, "Windows (SEH) Exception | ");
824
825 switch(code) {
826 case EXCEPTION_ACCESS_VIOLATION:
827 appendTxt(cTxtBuf, cTxtPos, "Access Violation (Invalid Memory Read/Write/Execute)");
828 break;
829 case EXCEPTION_ARRAY_BOUNDS_EXCEEDED:
830 appendTxt(cTxtBuf, cTxtPos, "Array Bounds Exceeded (Bounds-check fault)");
831 break;
832 case EXCEPTION_FLT_DIVIDE_BY_ZERO:
833 appendTxt(cTxtBuf, cTxtPos, "Float Divide By Zero");
834 break;
835 case EXCEPTION_INT_DIVIDE_BY_ZERO:
836 appendTxt(cTxtBuf, cTxtPos, "Integer Divide By Zero");
837 break;
838 case EXCEPTION_ILLEGAL_INSTRUCTION:
839 appendTxt(cTxtBuf, cTxtPos, "Illegal Instruction (Invalid or unsupported CPU instruction)");
840 break;
841 case EXCEPTION_PRIV_INSTRUCTION:
842 appendTxt(cTxtBuf, cTxtPos, "Privileged Instruction (Privileged CPU instruction in user mode)");
843 break;
844 case EXCEPTION_DATATYPE_MISALIGNMENT:
845 appendTxt(cTxtBuf, cTxtPos, "Datatype Misalignment (Alignment fault)");
846 break;
847 default:
848 appendTxt(cTxtBuf, cTxtPos, "Unknown Code: ");
849 appendHex(cTxtBuf, cTxtPos, code);
850 break;
851 }
852
853 impLogWrite(cTxtBuf, cTxtPos, true);
854 return EXCEPTION_EXECUTE_HANDLER;
855 }

881 {
882 // Re-Entrancy Guard
883 if(::InterlockedExchange(&crashing, 1) != 0)
884 hardFailFast();
885
886 cTxtPos = 0;
887 cTxtBuf[0] = '\0';

906 }
907 }
908 catch(...) {
909 appendTxt(cTxtBuf, cTxtPos, "Formatting failure while handling terminate");
910 }

912 impLogWrite(cTxtBuf, cTxtPos, true);
913 hardFailFast();

955 [[noreturn]] void reraiseDefault(int signum) noexcept
956 {
957 if(std::signal(signum, SIG_DFL) == SIG_ERR)
958 hardFailFast();
959 (void)std::raise(signum);
960 // If the default action did not terminate (or raise returned for any reason),
961 hardFailFast();
962 }
963
964 [[noreturn]] void signalHandler(int signum) noexcept
965 {
966 static volatile LONG inSignal = 0;
967 if(::InterlockedExchange(&inSignal, 1) != 0)
968 hardFailFast();
969
970 // This handler is expected to be installed only for SIGABRT.
971 if(signum != SIGABRT)
972 reraiseDefault(signum);
973
974 // Known "fatal log -> abort()" path: do not write crash log, just let default happen.
975 if(::InterlockedCompareExchange(&fatalLog, 0, 0) != 0)
976 reraiseDefault(SIGABRT);
977
978 // Absolute recursion stopper: any re-entry => immediate termination.
979 if(::InterlockedExchange(&crashing, 1) != 0)
980 hardFailFast();
981
982 cTxtPos = 0;
983 cTxtBuf[0] = '\0';
984 appendTxt(cTxtBuf, cTxtPos, "C Runtime Signal | [");
985 appendDec(cTxtBuf, cTxtPos, signum);
986 appendTxt(cTxtBuf, cTxtPos, "] SIGABRT: Abnormal termination (e.g. abort() or failed assert).");
987 impLogWrite(cTxtBuf, cTxtPos, true);
988
989 // Preserve default abort semantics (WER/debugger/etc.).
990 reraiseDefault(SIGABRT);
991 }

994 {

1624 // Generic helper to handle assignment with left-truncation
1625 template <typename LenT, size_t N>
1626 constexpr void assign_(char (&buf)[N], LenT& lenOut, std::string_view sv) noexcept
1627 {
1628 if(sv.empty()) return;
1629
1630 size_t beg{0};
1631 size_t cpy = sv.size();
1632 constexpr size_t MaxLen = N - 1; // Reserve 1 for null terminator
1633
1634 if(cpy > MaxLen) {
1635 buf[0] = '~';
1636 buf[MaxLen] = '\0';
1637 beg = 1;
1638 cpy = MaxLen - 1; // Truncate logic: Max capacity minus tilde
1639 lenOut = static_cast<LenT>(MaxLen);
1640 }
1641 else {
1642 buf[cpy] = '\0';
1643 lenOut = static_cast<LenT>(cpy);
1644 }
1645
1646 if(cpy > 0) {
1647 // Logic: copy the *end* of the string (suffix)
1648 const char* srcStart = sv.data() + sv.size() - cpy;
1649 if(std::is_constant_evaluated())
1650 std::copy_n(srcStart, cpy, buf + beg);
1651 else
1652 std::memcpy(buf + beg, srcStart, cpy);
1653 }

1666 const auto lastWin = s.rfind('\\');
1667 const auto lastUnix = s.rfind('/');
1668 const auto last = (lastWin == std::string_view::npos)
1669 ? lastUnix
1670 : (lastUnix == std::string_view::npos)
1671 ? lastWin
1672 : std::max(lastWin, lastUnix);
1673 const auto begin = (last == std::string_view::npos) ? 0 : last + 1;
1674 if(begin < s.size())
1675 assign_(fileName_, fileNameLen_, s.substr(begin));
1676 }
1677
1678 if(funcName && *funcName) {
1679
1680 size_t endIdx = 0;
1681 while(endIdx < 2048 && funcName[endIdx] != '\0' && funcName[endIdx] != '(')
1682 ++endIdx;
1683
1684 bool opeFound{false};
1685 size_t opeIdx = 0;
1686 for(size_t ii = 0; ii < endIdx; ++ii) {
1687 if(endIdx - ii < 8) break;
1688 if(funcName[ii + 0] == 'o' && funcName[ii + 1] == 'p' && funcName[ii + 2] == 'e' && funcName[ii + 3] == 'r' &&
1689 funcName[ii + 4] == 'a' && funcName[ii + 5] == 't' && funcName[ii + 6] == 'o' && funcName[ii + 7] == 'r') {
1690 const bool lftOk = (ii == 0) || isSpaceChar_(static_cast<unsigned char>(funcName[ii - 1])) || (funcName[ii - 1] == ':');
1691 const bool ritOk = ((ii + 8) == endIdx) || !isIdentifierChar_(static_cast<unsigned char>(funcName[ii + 8]));
1692 if(lftOk && ritOk) {
1693 opeFound = true;
1694 opeIdx = ii;
1695 }
1696 }
1697 }
1698
1699 size_t begIdx = 0;
1700
1701 size_t ignoreSpace = 0;
1702 size_t tmpIdx = opeFound ? opeIdx : endIdx;
1703 while(tmpIdx > 0) {
1704 --tmpIdx;
1705 if(funcName[tmpIdx] == '>') {
1706 ++ignoreSpace;
1707 }
1708 else if(funcName[tmpIdx] == '<') {
1709 if(ignoreSpace > 0)
1710 --ignoreSpace;
1711 }
1712 else if(ignoreSpace == 0 && isSpaceChar_(static_cast<unsigned char>(funcName[tmpIdx]))) {
1713 begIdx = tmpIdx + 1;
1714 break;
1715 }
1716 }
1717
1718 while(begIdx < endIdx && isSpaceChar_(static_cast<unsigned char>(funcName[begIdx])))

1720
1721 if(begIdx < endIdx) {

1724 }
1725 }

1986 {
1987 // 1. Configure OUTPUT Handle (STD_ERROR_HANDLE)
1988 // This handles text wrapping and newline processing.
1989 const HANDLE hOut = ::GetStdHandle(STD_ERROR_HANDLE);
1990 if(hOut == INVALID_HANDLE_VALUE || hOut == nullptr)
1991 return false;
1992
1993 // specific check: Is this a real console?
1994 // If h is a file redirect, GetConsoleMode fails.
1995 // We want to fail in that case because WriteConsoleW won't work on files.
1996 DWORD outMode = 0;
1997 if(::GetConsoleMode(hOut, &outMode) == 0)
1998 return false; // Not a real console
1999
2000 const DWORD desiredOutMode = outMode | ENABLE_PROCESSED_OUTPUT | ENABLE_WRAP_AT_EOL_OUTPUT;
2001 if(outMode != desiredOutMode)
2002 ::SetConsoleMode(hOut, desiredOutMode);
2003
2004 // 2. Configure INPUT Handle (STD_INPUT_HANDLE)
2005 // This handles Mouse Clicks and Quick Edit Mode.
2006 const HANDLE hIn = ::GetStdHandle(STD_INPUT_HANDLE);
2007 if(hIn != INVALID_HANDLE_VALUE && hIn != nullptr) {
2008 DWORD inMode = 0;
2009 if(::GetConsoleMode(hIn, &inMode)) {
2010 // ENABLE_EXTENDED_FLAGS is required to change Quick Edit settings
2011 DWORD desiredInMode = inMode | ENABLE_EXTENDED_FLAGS;
2012
2013 // DISABLE Quick Edit Mode (stops mouse clicks from pausing execution)
2014 desiredInMode &= ~ENABLE_QUICK_EDIT_MODE;
2015 // DISABLE Insert Mode
2016 desiredInMode &= ~ENABLE_INSERT_MODE;
2017 // DISABLE Mouse Input (Prevents accidental processing overhead)
2018 desiredInMode &= ~ENABLE_MOUSE_INPUT;
2019
2020 if(inMode != desiredInMode)
2021 ::SetConsoleMode(hIn, desiredInMode);
2022 }
2023 }
2024
2025 QString title{QString(npi::AppMetaData::Name_).append(" Console")};
2026 ::SetConsoleTitleW(reinterpret_cast<LPCWSTR>(title.utf16()));
2027
2028 return true;

2788 static void AppendDateTime_(BufferT& buffer, const std::chrono::system_clock::time_point& tp) noexcept
2789 {
2790 // Handle invalid/empty/default time points
2791 if(tp == std::chrono::system_clock::time_point::min()) {
2792 AppendChar_(buffer, "----.--.-- --:--:--.---", 23);
2793 return;
2794 }
2795
2796 // Floor to seconds and derive milliseconds from it
2797 const auto secTp = std::chrono::floor<std::chrono::seconds>(tp);
2798 const std::time_t sec = std::chrono::system_clock::to_time_t(secTp);
2799
2800 // Thread local "cached" (holds "YYYY-MM-DD HH:MM:SS" + null = 20 bytes)
2801 thread_local bool haveCache = false;
2802 thread_local std::time_t lastSec = 0;
2803 thread_local char cached[20] = {}; // 19 + \0
2804
2805 // Only update cache if the second has changed
2806 if(sec != lastSec || !haveCache) {
2807 std::tm bt{};
2808
2809 // `localtime_s` returns 0 on success
2810 const int ok = ::localtime_s(&bt, &sec);
2811
2812 // `std::snprintf` returns no. of characters written
2813 const int written = (ok == 0)
2814 ? std::snprintf(cached, sizeof(cached),
2815 "%04d.%02d.%02d %02d:%02d:%02d",
2816 bt.tm_year + 1900, bt.tm_mon + 1, bt.tm_mday,
2817 bt.tm_hour, bt.tm_min, bt.tm_sec)
2818 : -1;
2819
2820 // On error append zeros and return
2821 if(written != 19) {
2822 AppendChar_(buffer, "----.--.-- --:--:--.---", 23);
2823 return;
2824 }
2825
2826 // On success update the thread local values
2827 lastSec = sec;
2828 haveCache = true;
2829 }
2830
2831 // Append the cached date/time (19 chars)
2832 AppendChar_(buffer, cached, 19);
2833
2834 // Get milliseconds from the time point and floored seconds
2835 qint64 msPart = std::chrono::duration_cast<std::chrono::milliseconds>(tp - secTp).count();
2836 if(msPart < 0) msPart = 0; // defensive
2837 if(msPart > 999) msPart %= 1000; // defensive
2838 const int ms = static_cast<int>(msPart);
2839
2840 // Append milliseconds
2841 char msBuf[4];
2842 msBuf[0] = '.';
2843 msBuf[1] = static_cast<char>('0' + (ms / 100));
2844 msBuf[2] = static_cast<char>('0' + ((ms / 10) % 10));
2845 msBuf[3] = static_cast<char>('0' + (ms % 10));
2846 AppendChar_(buffer, msBuf, 4);
2847 }
