20 * \brief Windows-specific routine for resolving paths.
21 \details This anonymous namespace contains Windows-specific routine for resolving Known Folders
22 * to absolute paths as QString values.
23 */
24namespace { // internal helpers
25#ifdef _WIN32
26 /**
27 * \brief Retrieve a Windows Known Folder absolute path as a QString.
28 *
29 * \param id
30 * The REFKNOWNFOLDERID of the requested folder (e.g. FOLDERID_ProgramData,
31 * FOLDERID_PublicDocuments).
32 *
33 * \return
34 * Absolute path on success; an empty QString on failure (including an
35 * HRESULT failure from SHGetKnownFolderPath or a null pointer result).
36 *
37 * \details
38 * Invokes SHGetKnownFolderPath with KF_FLAG_DEFAULT to obtain the filesystem
39 * path for the specified Known Folder. On success, the shell allocates a
40 * wide-character buffer; this function converts it to QString and then
41 * releases the buffer via CoTaskMemFree. No exceptions are thrown; errors
42 * are reported by returning an empty QString.
43 *
44 * \note
45 * - Available only on Windows builds (_WIN32).
46 * - The returned QString is a value; callers incur normal implicit-sharing
47 * refcount costs.
48 * - The function is noexcept: it does not throw exceptions. Callers are
49 * responsible for checking for an empty QString result and handling errors.
50 *
51 * \par Error handling
52 * - If SHGetKnownFolderPath returns a failure HRESULT (FAILED(hr)) or a null
53 * output pointer, this function returns an empty QString.
54 *
55 * \par Thread safety
56 * - The function is thread-safe: it uses no shared mutable state.
57 *
58 * \sa SHGetKnownFolderPath, FOLDERID_ProgramData, FOLDERID_PublicDocuments
59 */
60 QString knownFolderPathWin(REFKNOWNFOLDERID id) noexcept {
61 PWSTR p = nullptr;
62 const HRESULT hr = SHGetKnownFolderPath(id, KF_FLAG_DEFAULT, nullptr, &p);
63 if(FAILED(hr) || !p)
64 return {};
65 QString out = QString::fromWCharArray(p);
66 CoTaskMemFree(p);
67 return out;
68 }
69#endif // _WIN32
70} // unnamed namespace
