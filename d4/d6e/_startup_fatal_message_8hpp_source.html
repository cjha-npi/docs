36 /// \brief Emergency startup fatal error message box.
37 /// \details This function displays a modal Windows-based message box
38 /// for use during application startup fatal errors. Since during
39 /// startup the QApplication and logging functionality may not yet
40 /// be initialized, this function uses the Windows API directly to
41 /// show the message box.
42 /// \param title The title of the message box, "[AppName] Fatal Error:" is prepended to this.
43 /// \param msg The message to display in the message box.
44 /// \warning This function does not return; it calls `std::abort()` after
45 /// displaying the message box which leads to abnormal program termination.
46 [[noreturn]] inline void startupFatalMessage(const QString& title, const QString& msg) noexcept
47 {
48 QString fullTitle = QString::fromLatin1(npi::AppMetaData::Name_) + QStringLiteral(" Fatal Error");
49 if(!title.isEmpty())
50 fullTitle.append(QStringLiteral(": ")).append(title);
51 const std::wstring t = fullTitle.toStdWString();
52 const std::wstring m = msg.toStdWString();
53 ::MessageBoxW(nullptr, m.c_str(), t.c_str(), MB_OK | MB_ICONERROR | MB_TOPMOST | MB_SETFOREGROUND);
54 std::abort();
55 }
56} // namespace npi::app
