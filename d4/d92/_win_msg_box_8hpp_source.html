3/**
4 * \file WinMsgBox.hpp
5 * \brief Windows message box helpers using the Unicode API.
6 *
7 * \details
8 * This header provides thin, dependency-light helpers around the Win32
9 * MessageBoxW function. They can be called safely even if no Qt GUI
10 * infrastructure (QApplication) has been created yet, making them suitable
11 * for early-startup, late-shutdown, and fatal-error reporting paths.
12 *
13 * Design goals:
14 * - Always use the Unicode (wide) Windows API to avoid code page issues.
15 * - Keep behavior predictable: choose an appropriate modality when the
16 * caller does not specify one, and foreground the dialog when no owner
17 * is available to reduce Z-order surprises.
18 * - Overload to accept optional QWidget owner, converting to HWND.
19 * - Avoid throwing exceptions; all functions are noexcept.
20 *
21 * Notes:
22 * - NOMINMAX is defined to prevent macro collisions with std::min/std::max.
23 * - WIN32_LEAN_AND_MEAN is defined to reduce compile-time overhead.
24 */

48 /// \brief Show a Windows message box, parented to an optional QWidget.
49 /// \details
50 /// Converts the caption and body to UTF-16 and invokes MessageBoxW.
51 /// If an owner widget is provided, its top-level window becomes the
52 /// owner. If no owner is available, the function strengthens modality
53 /// to task-modal and requests foreground activation to reduce the risk
54 /// of the dialog appearing behind other windows.
55 ///
56 /// Modality policy:
57 /// - If the caller does not supply any MB_* modality flags, the function
58 /// applies MB_APPLMODAL when an owner HWND exists, otherwise MB_TASKMODAL.
59 /// - If the caller supplies a modality flag, it is honored as-is.
60 ///
61 /// \param wid Optional QWidget whose top-level window will own the dialog.
62 /// May be nullptr; child widgets are accepted.
63 /// \param title Caption text (QString).
64 /// \param msg Body text (QString).
65 /// \param flags Combination of MB_* flags (icons, buttons, modality, etc.).
66 /// \remarks
67 /// - Calling winId() can materialize a native window for the owner, which
68 /// is typically desired for correct parenting and Z-order behavior.
69 /// - The function does not return a button ID. If the result is needed,
70 /// introduce an overload that returns the int value of MessageBoxW.
71 inline void msgBox(const QWidget* wid,
72 const QString& title,
73 const QString& msg,
74 UINT flags) noexcept
75 {
76 HWND h = nullptr;
77 if(wid) {
78 const QWidget* top = wid->window(); // Top-level QWidget used for ownership.
79 const WId id = top->winId(); // Ensures a native handle exists.
80 if(id) {
81 h = reinterpret_cast<HWND>(id); // Owner HWND for MessageBoxW.
82 }
83 }
84
85 const std::wstring wTitle = title.toStdWString(); // UTF-16 caption for MessageBoxW.
86 const std::wstring wMsg = msg.toStdWString(); // UTF-16 body for MessageBoxW.
87
88 constexpr UINT kModalityMask = MB_SYSTEMMODAL | MB_TASKMODAL | MB_APPLMODAL; // Bit mask of all modality flags.
89 UINT effFlags = flags; // Effective flags passed to MessageBoxW.
90
91 if((effFlags & kModalityMask) == 0)
92 effFlags |= (h ? MB_APPLMODAL : MB_TASKMODAL); // Auto-modality when unspecified.
93
94 if(!h)
95 effFlags |= MB_SETFOREGROUND; // Reduce Z-order issues without owner.
96
97 ::MessageBoxW(h, wMsg.c_str(), wTitle.c_str(), effFlags);
98 }
