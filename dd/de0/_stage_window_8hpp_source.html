1/**
2* \file StageWindow.hpp
3* \brief Header-only utility for showing a top-level widget either instantly or with a source-origin animation.
4*
5* \details
6* This header defines `npi::StageWindow`, a helper class for top-level `QWidget` display:
7* - `quick_()` positions and shows immediately.
8* - `animate_()` creates a temporary frameless overlay window that grows/moves from a source
9* rectangle (cursor or a "caller" widget) to the target widget's frame geometry, while the
10* real widget fades from 0.0 to 1.0 opacity at the end.
11*
12* Positioning guarantees:
13* - The target widget is adjusted to remain inside the available geometry of the chosen screen.
14* - Optional centering over a "center" widget.
15*
16* Safety and lifetime notes:
17* - `animate_()` does not capture `this`; it captures a `QPointer<QWidget>` to avoid use-after-free.
18* - If the target widget is deleted during the animation, the temporary overlay is still cleaned up.
19*
20* \warning Call these APIs from the GUI thread. Showing/moving widgets from worker threads is undefined.
21*/

144 /// \brief Show with an animated transition from a small source rect (cursor or caller) to the target.
145 /// \details
146 /// Steps:
147 /// 1. Compute `begRect` from `caller_` or a tiny rect at `QCursor::pos()`.
148 /// 2. Call `positionWidget_()` to place within screen bounds and center if requested.
149 /// 3. Set target widget opacity to 0.0 and show it using `callWidgetCorrectShowMethod_()`.
150 /// 4. Create a temporary frameless tool window (overlay) at `begRect` with a background color derived
151 /// from the target widget palette (slightly lightened/darkened for contrast).
152 /// 5. Animate the overlay geometry from `begRect` to the target frame geometry.
153 /// 6. On finish, set target opacity to 1.0 if still alive and delete the overlay.
154 /// \note
155 /// - The overlay is `Qt::Tool` and `WA_TransparentForMouseEvents` to avoid focus/taskbar issues.
156 /// - The animation object uses `DeleteWhenStopped` for cleanup.
157 /// - The lambda does not capture `this` to avoid lifetime hazards.
158 void animate_() const
159 {
160 if(widget_.isNull() || !widget_->isWindow())
161 return;
162
163 // Determine starting rect for the overlay animation.
164 const QRect begRect{
165 caller_.isNull()
166 ? QRect{QCursor::pos().x() - 1, QCursor::pos().y() - 1, 3, 3}
167 : (caller_->isWindow()
168 ? caller_->frameGeometry()
169 : QRect{caller_->mapToGlobal(QPoint{0, 0}), caller_->size()})
170 };
171
172 // Position target and compute its final frame geometry now.
173 positionWidget_();
174
175 // Make target invisible first, then show it correctly so frameGeometry() is valid for the end rect.
176 widget_->setWindowOpacity(0.0);
177 callWidgetCorrectShowMethod_();
178
179 // Guard ensures we do not leave the widget transparent on early exit.
180 OpacityGuard guard{widget_};
181
182 // Derive overlay background color for a subtle contrast flash.
183 QColor bgColor{widget_->palette().color(QPalette::Window)};
184 const int avg = (bgColor.red() + bgColor.green() + bgColor.blue()) / 3;
185 bgColor = (avg > 127) ? bgColor.darker(120) : bgColor.lighter(120);
186
187 // Create temporary overlay; no parent so it is its own top-level.
188 QWidget* temp = new QWidget;
189 temp->setWindowFlags(Qt::FramelessWindowHint | Qt::Tool);
190 temp->setAttribute(Qt::WA_StyledBackground, true);
191 temp->setAttribute(Qt::WA_TransparentForMouseEvents, true);
192 temp->setGeometry(begRect);
193 temp->setStyleSheet(QString("background-color: %1;").arg(bgColor.name()));
194
195 // Determine duration.
196 int dura = duration_;
197 if(dura == -1) { // Use auto-duration heuristic
198 const QPoint diff = begRect.topLeft() - widget_->frameGeometry().topLeft();
199 // Heuristic: base plus path-length factor, clamped to [MinDura_, MaxDura_].
200 const int term = static_cast<int>(qSqrt((diff.x() * diff.x()) + (diff.y() * diff.y())) / 4);
201 dura = qMin(MaxDura_, MinDura_ + term);
202 }
203
204 // Animate overlay geometry.
205 QPropertyAnimation* amin = new QPropertyAnimation(temp, "geometry");
206 amin->setDuration(dura);
207 amin->setStartValue(begRect);
208 amin->setEndValue(widget_->frameGeometry());
209 amin->setEasingCurve(easingCurve_);
210
211 // Capture a safe handle to the widget to avoid dereferencing a dead `this`.
212 QPointer<QWidget> target = widget_;
213
214 QObject::connect(amin, &QPropertyAnimation::finished, [temp, target]() {
215 if(!target.isNull())
216 target->setWindowOpacity(1.0);
217 temp->deleteLater();
218 });
219
220 temp->show();
221 amin->start(QAbstractAnimation::DeleteWhenStopped);
222
223 // At this point the normal path hands off restoration to the finished-callback.
224 // Disable the guard so it does not undo the fade.
225 guard.release_();
226 }

240 /// \brief Compute and apply an on-screen position for `widget_`, respecting frames/title bars.
241 /// \details
242 /// Logic:
243 /// - Estimate full framed size if not yet shown (geometry == frameGeometry).
244 /// - If `center_` is set, center over it (using frame geometry when it is a window).
245 /// - Else, if the widget has never been displayed and sits at (0,0), center on the screen under the cursor.
246 /// - Select screen using the frame-geometry center of `widget_` (or `center_`); fallback to primary.
247 /// - Clamp position to available geometry (no overflow left/top/right/bottom).
248 void positionWidget_() const
249 {
250 if(widget_.isNull() || !widget_->isWindow())
251 return;
252
253 // Calculate the extra space needed for the title bar.
254 // If the widget has a title (WindowTitleHint is set), obtain the title bar height
255 const auto titleHeight{(widget_->windowFlags() & Qt::WindowTitleHint) ? qApp->style()->pixelMetric(QStyle::PM_TitleBarHeight) : 0};
256
257 // Calculate the extra space needed for the window frame.
258 // If the widget is frameless, set frameSize to 0; otherwise, use the default frame width.
259 const auto frameSize{(widget_->windowFlags() & Qt::FramelessWindowHint) ? 0 : qApp->style()->pixelMetric(QStyle::PM_DefaultFrameWidth)};
260
261 // Determine full framed size.
262 QSize widFullSize;
263 if(widget_->geometry() == widget_->frameGeometry()) {
264 // Not shown yet; synthesize framed size from client size.
265 widFullSize = QSize{
266 widget_->size().width() + (2 * frameSize),
267 widget_->size().height() + titleHeight + (2 * frameSize)
268 };
269 }
270 else {
271 widFullSize = widget_->frameGeometry().size();
272 }
273
274 // Start from current position.
275 QPoint widPos{widget_->pos()};
276
277 // Choose a screen reference.
278 QScreen* screen{qApp->screenAt(widget_->frameGeometry().center())};
279
280 if(!center_.isNull()) {
281 // Center over `center_`.
282 screen = qApp->screenAt(center_->geometry().center());
283 const QRect cenRect = center_->isWindow() ? center_->frameGeometry() : center_->geometry();
284 const int widPosX = cenRect.x() + (cenRect.width() - widFullSize.width()) / 2;
285 const int widPosY = cenRect.y() + (cenRect.height() - widFullSize.height()) / 2;
286 widPos = QPoint{widPosX, widPosY};
287 }
288 else if(widget_->pos() == QPoint{0, 0} && widget_->geometry() == widget_->frameGeometry()) {
289 // Likely first show and no explicit centering widget; center on cursor's screen.
290 screen = qApp->screenAt(QCursor::pos());
291 if(screen) {
292 const QRect avail = screen->availableGeometry();
293 const int widPosX = avail.center().x() - (widFullSize.width() / 2);
294 const int widPosY = avail.center().y() - (widFullSize.height() / 2);
295 widPos = QPoint{widPosX, widPosY};
296 }
297 }
298
299 if(screen == nullptr)
300 screen = qApp->primaryScreen();
301
302 const QRect avail = screen->availableGeometry();
303
304 // Horizontal clamp.
305 if(widPos.x() + widFullSize.width() > avail.x() + avail.width())
306 widPos.setX(avail.x() + avail.width() - widFullSize.width());
307 if(widPos.x() < avail.x())
308 widPos.setX(avail.x());
309
310 // Vertical clamp.
311 if(widPos.y() + widFullSize.height() > avail.y() + avail.height())
312 widPos.setY(avail.y() + avail.height() - widFullSize.height());
313 if(widPos.y() < avail.y())
314 widPos.setY(avail.y());
315
316 widget_->move(widPos);
317 }

319 /// \brief Show the widget in a state-correct way.
320 /// \details
321 /// - If minimized: call `showNormal()`.
322 /// - Else: if visible, `hide()` first to reset painting, then `show()`.
323 /// - Else: `show()`.
324 void callWidgetCorrectShowMethod_() const
325 {
326 if(widget_.isNull() || !widget_->isWindow())
327 return;
328
329 if(widget_->isMinimized()) {
330 widget_->showNormal();
331 }
332 else {
333 if(widget_->isVisible())
334 widget_->hide();
335 widget_->show();
336 }

339 /**
340 * \brief Helper for `StageWindow` class. A RAII guard that restores a widget's window opacity on scope exit.
341 * \details
342 * Use when you temporarily set a target widget's opacity to 0.0 before
343 * an animation is fully wired. If control leaves the scope early (exception
344 * or early return) and the guard is still active, it restores the widget to
345 * full opacity (1.0) in the destructor. After the animation has been started
346 * and its own finished-handler will handle restoration, call release_() to
347 * disable the guard.
348 * \note
349 * The guard holds a QPointer to avoid dereferencing a deleted widget. If the
350 * widget is destroyed before scope exit, no action is taken.
351 * \warning
352 * Construct and use from the GUI thread only. The opacity change affects GUI state.
353 */
354 struct OpacityGuard {
355 /// \brief QPointer to the guarded widget (may become null if the widget is deleted).
356 /// \details
357 /// QPointer auto-nullifies on QObject destruction, preventing use-after-free in
358 /// the destructor when attempting to restore opacity.
359 QPointer<QWidget> guardedWid_;
360
361 /// \brief Whether the guard is active.
362 /// \details
363 /// If true at destruction time and the widget pointer is valid, the guard will
364 /// restore opacity to 1.0. Calling release_() sets this to false to make the
365 /// destructor a no-op regarding opacity restoration.
366 bool active_{true};
367
368 /// \brief Construct a guard for the given widget.
369 /// \param w Pointer to the widget whose opacity should be restored on scope exit.
370 /// \post guardedWid_ points to @p w; active_ is true.
371 explicit OpacityGuard(QWidget* w) : guardedWid_{w} {}
372
373 /// \brief Destructor; restores opacity if still active and the widget is alive.
374 /// \details
375 /// If active_ is true and guardedWid_ is non-null, sets guardedWid_->setWindowOpacity(1.0).
376 /// If the widget was deleted during the guarded scope, guardedWid_ is null and no action is taken.
377 ~OpacityGuard()
378 {
379 if(active_ && !guardedWid_.isNull())
380 guardedWid_->setWindowOpacity(1.0);
381 }
382
383 /// \brief Disable the guard so the destructor does not restore opacity.
384 /// \details
385 /// Call this after you have successfully started your animation and arranged
386 /// for normal restoration (e.g., in the animation's finished handler).
387 /// \post active_ is false; the destructor will not modify widget opacity.
388 void release_() { active_ = false; }
389
390 OpacityGuard(const OpacityGuard&) = delete; ///< deleted copy constructor
391 OpacityGuard& operator=(const OpacityGuard&) = delete; ///< deleted copy assignment
392 OpacityGuard(OpacityGuard&&) = delete; ///< deleted move constructor
393 OpacityGuard& operator=(OpacityGuard&&) = delete; ///< deleted move assignment
394 };
395 };
