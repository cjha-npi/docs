42 /// \brief Absolute organization-level ProgramData path.
43 /// \details
44 /// Returns the absolute, normalized path to:
45 /// [ProgramData]/[OrganizationName]
46 /// Resolution order:
47 /// 1) Windows API (SHGetKnownFolderPath with FOLDERID_ProgramData)
48 /// 2) Environment variable fallback: PROGRAMDATA
49 /// Then:
50 /// - Verifies the base folder exists (fatal if missing).
51 /// - Ensures the organization subdirectory exists (creates it; fatal if it fails).
52 /// - Canonicalizes when possible (resolves junctions/symlinks); otherwise cleans the absolute path.
53 /// - Converts separators to forward slashes; UNC is returned as //server/share/....
54 ///
55 /// The result is cached in a function-local static and reused on subsequent calls.
56 ///
57 /// \return Reference to a process-lifetime QString with the absolute path using forward slashes.
58 /// \post The returned reference remains valid for the lifetime of the process.
59 /// \note
60 /// - Windows-only: Paths use forward slashes for Qt friendliness; call
61 /// QDir::toNativeSeparators(...) before passing to Win32 APIs.
62 /// - Multi-Thread: Safe, first-call initialization is thread-safe (C++11). Subsequent calls are lock-free.
63 /// \warning On unrecoverable failure (cannot locate or create the required directory),
64 /// a native dialog is shown and the process is terminated (no return).
65 /// \par Example
66 /// \code
67 /// // Build a file path within the org folder (Qt accepts forward slashes)
68 /// const QString orgDir = npi::app::orgProgramDataDir(); // e.g. C:/ProgramData/npi_electronic
69 /// const QString file = QDir(orgDir).filePath("global.log");
70 ///
71 /// // When calling Win32, convert once:
72 /// const QString native = QDir::toNativeSeparators(orgDir); // e.g. C:\ProgramData\npi_electronic
73 /// // CreateDirectoryW(reinterpret_cast<LPCWSTR>(native.utf16()), nullptr);
74 /// \endcode
75 [[nodiscard]] const QString& orgProgramDataDir() noexcept;
