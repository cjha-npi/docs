1/**
2* \file TextSpec.hpp
3* \brief Thread-safe font utility for application and fixed-width fonts.
4*
5* \details
6* This header defines npi::TextSpec, a small utility that centralizes font handling:
7* - Maintains a global application font and a global fixed-width font.
8* - Provides style flags (bold, italic, underline, overline, strikeout) and capitalization.
9* - Supports absolute and relative point sizes and computes effective sizes.
10* - Caches QFont and its metrics (QFontMetrics, QFontMetricsF) keyed by style+size.
11* - Uses per-family mutexes to guard static state and caches.
12*
13* \note GUI thread requirement:
14* Any interaction with QWidget or QApplication must occur on the GUI thread.
15* This class posts updates using queued connections where needed.
16*
17* \warning Ordering note for SetAppFont_ and SetFixFont_:
18* These methods update internal state and clear caches under a mutex, unlock,
19* and then perform GUI updates (qApp->setFont or posting QWidget::update()).
20* This introduces a brief window where cache state and GUI state may not be
21* perfectly in sync. This is acceptable in practice and avoids holding a lock
22* across GUI calls.
23*/

47 /**
48 * \brief Thread-safe font utility with style, sizing, and caching.
49 *
50 * \details
51 * The TextSpec class exposes:
52 * - Global setters for the base application font and the base fixed-width font.
53 * - A Style struct to describe per-request font attributes.
54 * - Absolute and relative point-size inputs with effective-size calculation.
55 * - Cached Groups of (QFont, QFontMetrics, QFontMetricsF) for fast reuse.
56 * - Static mutexes to protect shared state and caches.
57 *
58 * Typical use:
59 * \code
60 * // Initialize at app startup (GUI thread):
61 * npi::TextSpec::SetAppFont_(qApp->font());
62 *
63 * // On demand, configure style and size:
64 * npi::TextSpec::Style st;
65 * st.bold_ = true;
66 * npi::TextSpec f{st};
67 * f.setRelativeSize_(+1);
68 *
69 * // Fetch QFont or metrics:
70 * QFont font = f.appFont_();
71 * QFontMetrics fm = f.appFontMetrics_();
72 * \endcode
73 */

134 /// \brief Set the global application font (family, size, style).
135 ///
136 /// \details
137 /// - Clears the application font cache so new Groups reflect the new base font.
138 /// - Calls qApp->setFont(font) after releasing the internal mutex.
139 ///
140 /// \pre Must be invoked when QApplication exists. Should be invoked on the GUI thread.
141 /// \param font New application base font.
142 static void SetAppFont_(const QFont& font)
143 {
144 AppFontMutex_.lock(); // Lock for thread safety.
145 if(font != AppFont_) { // Only update if the new font differs.
146 AppFont_ = font; // Update the global application font.
147 AppFontSize_ = (font.pointSize() < 0 ? 9 : font.pointSize()); // Update the font size.
148 AppGroupCache_.clear(); // Clear the cache to ensure new font groups are created.
149 AppFontMutex_.unlock(); // Unlock before GUI call.
150 qApp->setFont(font); // Apply the new font to the application.
151 }
152 else {
153 AppFontMutex_.unlock(); // Unlock if no change.
154 }
155 }

157 /// \brief Set the global fixed-width font (family, size, style).
158 ///
159 /// \details
160 /// - Clears the fixed-width font cache.
161 /// - Queues `QWidget::update()` on widgets marked with `npi::GuiProps::FixedWidthFontWidget_`.
162 /// Lifetime is guarded via QPointer in the queued lambda.
163 ///
164 /// \pre Should be invoked on the GUI thread.
165 /// \param font New fixed-width base font.
166 static void SetFixFont_(const QFont& font)
167 {
168 FixFontMutex_.lock(); // Lock for thread safety.
169 if(font != FixFont_) { // Only update if the new font differs.
170 FixFont_ = font; // Update the global fixed-width font.
171 FixFontSize_ = (font.pointSize() < 0 ? 9 : font.pointSize()); // Update the font size.
172 FixGroupCache_.clear(); // Clear the fixed-width font cache.
173 FixFontMutex_.unlock(); // Unlock before GUI calls.
174
175 // Iterate over all widgets in the application.
176 const QList<QWidget*> widgets = qApp->allWidgets();
177 for(QWidget* widget : widgets) {
178 // If the widget uses the fixed-width font, force an update to apply the new font.
179 if(QPointer<QWidget> w{widget}; w && w->property(npi::GuiProps::FixedWidthFontWidget_).toBool()) {
180 QMetaObject::invokeMethod(qApp, [w] { if(w) w->update(); }, Qt::QueuedConnection);
181 }
182 }
183 }
184 else {
185 FixFontMutex_.unlock(); // Unlock if no change.
186 }
187 }

254 if(color == Color::Magenta)

462 /// \brief Compute a hash key from style and size.
463 ///
464 /// \details
465 /// Combines boolean flags, capitalization, and size into a single 32-bit integer.
466 /// Bit layout before qHash:
467 /// - bits 0..4 : bold, italic, underline, overline, strikeout
468 /// - bits 5..12 : capitalization (8 bits)
469 /// - bits 13..28: point size (16 bits, truncated, assumes size is within [0, 65535])
470 ///
471 /// \param style Style flags and capitalization.
472 /// \param size Point size used for the key (truncated to 16 bits).
473 /// \param seed Optional hash seed.
474 /// \return size_t hash suitable as a QHash key.
475 static size_t Hash_(const Style& style, const int size, size_t seed = 0)
476 {
477 quint32 combined = 0;
478 combined |= (style.bold_ ? 1u : 0u) << 0;
479 combined |= (style.italic_ ? 1u : 0u) << 1;
480 combined |= (style.underline_ ? 1u : 0u) << 2;
481 combined |= (style.overline_ ? 1u : 0u) << 3;
482 combined |= (style.strikeout_ ? 1u : 0u) << 4;
483 combined |= (static_cast<quint32>(style.capitalization_) & 0xFF) << 5;
484 combined |= (static_cast<quint32>(size) & 0xFFFF) << 13;
485 return qHash(combined, seed);
486 }

488 /// \brief Get or create the cached Group for the application font family.
489 ///
490 /// \details
491 /// Retrieves or creates a cached Group for the application font. The effective
492 /// font size is computed based on either the absolute size (if > 0) or the current
493 /// application font size plus the relative size adjustment. A hash key is computed
494 /// from the style and size, and the cache is checked for an existing entry. If not
495 /// found, a new QFont is configured, stored in a Group, and inserted into the cache.
496 ///
497 /// \param style Style flags and capitalization.
498 /// \param absSize Absolute point size (> 0 overrides).
499 /// \param relSize Relative point size delta.
500 /// \return Reference to cached Group (valid for the lifetime of the cache).
501 ///
502 /// \note Uses AppFont_ as the base family.
503 static const Group& GetAppGroup_(const Style& style, const int absSize, const int relSize)
504 {
505 const int size = absSize > 0 ? absSize : (AppFontSize_ + relSize > 0 ? AppFontSize_ + relSize : AppFontSize_); // Compute effective font size.
506 const size_t hash = Hash_(style, size); // Compute the cache key using the style and effective size.
507 auto it = AppGroupCache_.find(hash); // Look for the key in the application font cache.
508 if(it == AppGroupCache_.end()) { // Not found in cache: create a new QFont based on the current AppFont_.
509 QFont font{AppFont_};
510 font.setPointSize(size);
511 font.setBold(style.bold_);
512 font.setItalic(style.italic_);
513 font.setUnderline(style.underline_);
514 font.setOverline(style.overline_);
515 font.setStrikeOut(style.strikeout_);
516 font.setCapitalization(style.capitalization_);
517 AppGroupCache_.insert(hash, Group{font}); // Insert the new Group into the cache.
518 it = AppGroupCache_.find(hash); // Update iterator after insertion and retrieve for newly inserted group.
519 }
520 return it.value(); // Return the cached Group containing the configured QFont and its metrics.
521 }
