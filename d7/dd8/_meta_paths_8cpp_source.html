49 /// \brief Resolve base path using Known Folder ID or environment variable fallback.
50 /// \details This functions attempts to resolve the base path for the given
51 /// folder ID by first querying the Windows API and falling back to an
52 /// environment variable if necessary.
53 /// \param folderId The Known Folder ID to query via SHGetKnownFolderPath.
54 /// \param envVar The environment variable to check if the API call fails.
55 /// \param envSuffix The suffix to append to the environment variable path.
56 /// \param pathNameForErrors The name of the path for error messages.
57 /// \warning On unrecoverable failure (cannot locate or validate the directory),
58 /// a native dialog is shown and the process is terminated (no return).
59 QString basePath(const KNOWNFOLDERID& folderId, const char* envVar, const QString& envSuffix, const QString& pathNameForErrors) noexcept
60 {
61 // String to hold the resolved base path.
62 QString base;
63
64 // Windows API attempt.
65 PWSTR p = nullptr;
66 const HRESULT hr = SHGetKnownFolderPath(folderId, KF_FLAG_DEFAULT, nullptr, &p);
67 if(SUCCEEDED(hr) && p) {
68 base = QString::fromWCharArray(p);
69 CoTaskMemFree(p);
70 }
71
72 // Environment variable fallback.
73 if(base.isEmpty()) {
74 if(const QString envPath = qEnvironmentVariable(envVar); !envPath.isEmpty()) {
75 base = envSuffix.isEmpty() ? envPath : QDir(envPath).filePath(envSuffix);
76 }
77 }
78
79 // Abort on failure to locate.
80 if(base.isEmpty()) {
81 npi::app::nativeFatalMessage(
82 QString("%1 Not Found").arg(pathNameForErrors),
83 QString("Could not locate the system's %1 directory. This folder is "
84 "required for application settings.")
85 .arg(pathNameForErrors));
86 }
87
88 // Abort if the path is missing or not a directory.
89 if(const QFileInfo info(base); !info.exists() || !info.isDir()) {
90 npi::app::nativeFatalMessage(
91 QString("%1 Invalid").arg(pathNameForErrors),
92 QString("The system's %1 path '%2' is missing or is not a "
93 "directory. Please ensure this folder exists.")
94 .arg(pathNameForErrors, QDir::fromNativeSeparators(base)));
95 }
96
97 return base;
98 }

100 /// \brief Create subdirectory under base path.
101 /// \details This function attempts to create a subdirectory
102 /// under the specified base path. On failure, a fatal error
103 /// message is displayed and the process is terminated.
104 /// \param base The base directory under which to create the subdirectory.
105 /// \param dirName The name of the subdirectory to create.
106 /// \warning On unrecoverable failure to create the directory,
107 /// a native dialog is shown and the process is terminated (no return).
108 QString createSubDir(const QString& base, const QString& dirName) noexcept
109 {
110 QDir root{base};
111 const QString target = root.filePath(dirName);
112 if(!root.mkpath(dirName)) {
113 npi::app::nativeFatalMessage(
114 "Cannot Create Directory",
115 QString("The required settings directory could not be created:\n%1\n\n"
116 "This may be due to a permissions issue or because a file "
117 "with the same name already exists.")
118 .arg(QDir::fromNativeSeparators(target)));
119 }
120 return QDir::cleanPath(target);
121 }
