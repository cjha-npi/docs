1/**
2 * @file Show.hpp
3 * @brief Header-only utility for showing a top-level widget either instantly or with a source-origin animation.
4 *
5 * @details
6 * This header defines a small helper class `npi::Show` for top-level `QWidget` display:
7 * - `quick_()` positions and shows immediately.
8 * - `animate_()` creates a temporary frameless overlay window that grows/moves from a source
9 * rectangle (cursor or a "caller" widget) to the target widget's frame geometry, while the
10 * real widget fades from 0.0 to 1.0 opacity at the end.
11 *
12 * Positioning guarantees:
13 * - The target widget is adjusted to remain inside the available geometry of the chosen screen.
14 * - Optional centering over a "center" widget.
15 *
16 * Safety and lifetime notes:
17 * - `animate_()` does not capture `this`; it captures a `QPointer<QWidget>` to avoid use-after-free.
18 * - If the target widget is deleted during the animation, the temporary overlay is still cleaned up.
19 *
20 * @warning Call these APIs from the GUI thread. Showing/moving widgets from worker threads is undefined.
21 */
22

164 /**
165 * @brief Show with an animated transition from a small source rect (cursor or caller) to the target.
166 *
167 * @details
168 * Steps:
169 * 1. Compute `begRect` from `caller_` or a tiny rect at `QCursor::pos()`.
170 * 2. Call `positionWidget_()` to place within screen bounds and center if requested.
171 * 3. Set target widget opacity to 0.0 and show it using `callWidgetCorrectShowMethod_()`.
172 * 4. Create a temporary frameless tool window (overlay) at `begRect` with a background color derived
173 * from the target widget palette (slightly lightened/darkened for contrast).
174 * 5. Animate the overlay geometry from `begRect` to the target frame geometry.
175 * 6. On finish, set target opacity to 1.0 if still alive and delete the overlay.
176 *
177 * @note
178 * - The overlay is `Qt::Tool` and `WA_TransparentForMouseEvents` to avoid focus/taskbar issues.
179 * - The animation object uses `DeleteWhenStopped` for cleanup.
180 * - The lambda does not capture `this` to avoid lifetime hazards.
181 */
182 void animate_() const {
183 if(widget_.isNull() || !widget_->isWindow())
184 return;
185
186 // Determine starting rect for the overlay animation.
187 const QRect begRect{
188 caller_.isNull()
189 ? QRect{QCursor::pos().x() - 1, QCursor::pos().y() - 1, 3, 3}
190 : (caller_->isWindow()
191 ? caller_->frameGeometry()
192 : QRect{caller_->mapToGlobal(QPoint{0, 0}), caller_->size()})
193 };
194
195 // Position target and compute its final frame geometry now.
196 positionWidget_();
197
198 // Make target invisible first, then show it correctly so frameGeometry() is valid for the end rect.
199 widget_->setWindowOpacity(0.0);
200 callWidgetCorrectShowMethod_();
201
202 // Guard ensures we do not leave the widget transparent on early exit.
203 OpacityGuard guard{widget_};
204
205 // Derive overlay background color for a subtle contrast flash.
206 QColor bgColor{widget_->palette().color(QPalette::Window)};
207 const int avg = (bgColor.red() + bgColor.green() + bgColor.blue()) / 3;
208 bgColor = (avg > 127) ? bgColor.darker(120) : bgColor.lighter(120);
209
210 // Create temporary overlay; no parent so it is its own top-level.
211 QWidget* temp = new QWidget;
212 temp->setWindowFlags(Qt::FramelessWindowHint | Qt::Tool);
213 temp->setAttribute(Qt::WA_StyledBackground, true);
214 temp->setAttribute(Qt::WA_TransparentForMouseEvents, true);
215 temp->setGeometry(begRect);
216 temp->setStyleSheet(QString("background-color: %1;").arg(bgColor.name()));
217
218 // Determine duration.
219 int dura = duration_;
220 if(dura < 10 || dura > 10000) {
221 const QPoint diff = begRect.topLeft() - widget_->frameGeometry().topLeft();
222 // Heuristic: base plus path-length factor, clamped to [MinDura_, MaxDura_].
223 const int term = static_cast<int>(qSqrt((diff.x() * diff.x()) + (diff.y() * diff.y())) / 4);
224 dura = qMin(MaxDura_, MinDura_ + term);
225 }
226
227 // Animate overlay geometry.
228 QPropertyAnimation* amin = new QPropertyAnimation(temp, "geometry");
229 amin->setDuration(dura);
230 amin->setStartValue(begRect);
231 amin->setEndValue(widget_->frameGeometry());
232 amin->setEasingCurve(easingCurve_);
233
234 // Capture a safe handle to the widget to avoid dereferencing a dead `this`.
235 QPointer<QWidget> target = widget_;
236
237 QObject::connect(amin, &QPropertyAnimation::finished, [temp, target]() {
238 if(!target.isNull())
239 target->setWindowOpacity(1.0);
240 temp->deleteLater();
241 });
242
243 temp->show();
244 amin->start(QAbstractAnimation::DeleteWhenStopped);
245
246 // At this point the normal path hands off restoration to the finished-callback.
247 // Disable the guard so it does not undo the fade.
248 guard.release_();
249 }

264 /**
265 * @brief Compute and apply an on-screen position for `widget_`, respecting frames/title bars.
266 *
267 * @details
268 * Logic:
269 * - Estimate full framed size if not yet shown (geometry == frameGeometry).
270 * - If `center_` is set, center over it (using frame geometry when it is a window).
271 * - Else, if the widget has never been displayed and sits at (0,0), center on the screen under the cursor.
272 * - Select screen using the frame-geometry center of `widget_` (or `center_`); fallback to primary.
273 * - Clamp position to available geometry (no overflow left/top/right/bottom).
274 */
275 void positionWidget_() const {
276 if(widget_.isNull() || !widget_->isWindow())
277 return;
278
279 // Calculate the extra space needed for the title bar.
280 // If the widget has a title (WindowTitleHint is set), obtain the title bar height
281 const auto titleHeight{(widget_->windowFlags() & Qt::WindowTitleHint) ? qApp->style()->pixelMetric(QStyle::PM_TitleBarHeight) : 0};
282
283 // Calculate the extra space needed for the window frame.
284 // If the widget is frameless, set frameSize to 0; otherwise, use the default frame width.
285 const auto frameSize{(widget_->windowFlags() & Qt::FramelessWindowHint) ? 0 : qApp->style()->pixelMetric(QStyle::PM_DefaultFrameWidth)};
286
287 // Determine full framed size.
288 QSize widFullSize;
289 if(widget_->geometry() == widget_->frameGeometry()) {
290 // Not shown yet; synthesize framed size from client size.
291 widFullSize = QSize{
292 widget_->size().width() + (2 * frameSize),
293 widget_->size().height() + titleHeight + (2 * frameSize)
294 };
295 }
296 else {
297 widFullSize = widget_->frameGeometry().size();
298 }
299
300 // Start from current position.
301 QPoint widPos{widget_->pos()};
302
303 // Choose a screen reference.
304 QScreen* screen{qApp->screenAt(widget_->frameGeometry().center())};
305
306 if(!center_.isNull()) {
307 // Center over `center_`.
308 screen = qApp->screenAt(center_->geometry().center());
309 const QRect cenRect = center_->isWindow() ? center_->frameGeometry() : center_->geometry();
310 const int widPosX = cenRect.x() + (cenRect.width() - widFullSize.width()) / 2;
311 const int widPosY = cenRect.y() + (cenRect.height() - widFullSize.height()) / 2;
312 widPos = QPoint{widPosX, widPosY};
313 }
314 else if(widget_->pos() == QPoint{0, 0} && widget_->geometry() == widget_->frameGeometry()) {
315 // Likely first show and no explicit centering widget; center on cursor's screen.
316 screen = qApp->screenAt(QCursor::pos());
317 if(screen) {
318 const QRect avail = screen->availableGeometry();
319 const int widPosX = avail.center().x() - (widFullSize.width() / 2);
320 const int widPosY = avail.center().y() - (widFullSize.height() / 2);
321 widPos = QPoint{widPosX, widPosY};
322 }
323 }
324
325 if(screen == nullptr)
326 screen = qApp->primaryScreen();
327
328 const QRect avail = screen->availableGeometry();
329
330 // Horizontal clamp.
331 if(widPos.x() + widFullSize.width() > avail.x() + avail.width())
332 widPos.setX(avail.x() + avail.width() - widFullSize.width());
333 if(widPos.x() < avail.x())
334 widPos.setX(avail.x());
335
336 // Vertical clamp.
337 if(widPos.y() + widFullSize.height() > avail.y() + avail.height())
338 widPos.setY(avail.y() + avail.height() - widFullSize.height());
339 if(widPos.y() < avail.y())
340 widPos.setY(avail.y());
341
342 widget_->move(widPos);
343 }

345 /**
346 * @brief Show the widget in a state-correct way.
347 *
348 * @details
349 * - If minimized: call `showNormal()`.
350 * - Else: if visible, `hide()` first to reset painting, then `show()`.
351 * - Else: `show()`.

353 void callWidgetCorrectShowMethod_() const {
354 if(widget_.isNull() || !widget_->isWindow())
355 return;
356
357 if(widget_->isMinimized()) {
358 widget_->showNormal();
359 }
360 else {
361 if(widget_->isVisible())
362 widget_->hide();
363 widget_->show();
364 }

367 /**
368 * @brief Helper for `Show` class. A RAII guard that restores a widget's window opacity on scope exit.
369 *
370 * @details
371 * Use when you temporarily set a target widget's opacity to 0.0 before
372 * an animation is fully wired. If control leaves the scope early (exception
373 * or early return) and the guard is still active, it restores the widget to
374 * full opacity (1.0) in the destructor. After the animation has been started
375 * and its own finished-handler will handle restoration, call release_() to
376 * disable the guard.
377 *
378 * @note
379 * The guard holds a QPointer to avoid dereferencing a deleted widget. If the
380 * widget is destroyed before scope exit, no action is taken.
381 *
382 * @warning
383 * Construct and use from the GUI thread only. The opacity change affects GUI state.
384 */
385 struct OpacityGuard {
386 /**
387 * @brief QPointer to the guarded widget (may become null if the widget is deleted).
388 *
389 * @details
390 * QPointer auto-nullifies on QObject destruction, preventing use-after-free in
391 * the destructor when attempting to restore opacity.
392 */
393 QPointer<QWidget> guardedWid_;
394
395 /**
396 * @brief Whether the guard is active.
397 *
398 * @details
399 * If true at destruction time and the widget pointer is valid, the guard will
400 * restore opacity to 1.0. Calling release_() sets this to false to make the
401 * destructor a no-op regarding opacity restoration.
402 */
403 bool active_{true};
404
405 /**
406 * @brief Construct a guard for the given widget.
407 * @param w Pointer to the widget whose opacity should be restored on scope exit.
408 *
409 * @post guardedWid_ points to @p w; active_ is true.
410 */
411 explicit OpacityGuard(QWidget* w) : guardedWid_{w} {}
412
413 /**
414 * @brief Destructor; restores opacity if still active and the widget is alive.
415 *
416 * @details
417 * If active_ is true and guardedWid_ is non-null, sets guardedWid_->setWindowOpacity(1.0).
418 * If the widget was deleted during the guarded scope, guardedWid_ is null and no action is taken.
419 */
420 ~OpacityGuard() {
421 if(active_ && !guardedWid_.isNull())
422 guardedWid_->setWindowOpacity(1.0);
423 }
424
425 /**
426 * @brief Disable the guard so the destructor does not restore opacity.
427 *
428 * @details
429 * Call this after you have successfully started your animation and arranged
430 * for normal restoration (e.g., in the animation's finished handler).
431 *
432 * @post active_ is false; the destructor will not modify widget opacity.
433 */
434 void release_() { active_ = false; }
435
436 OpacityGuard(const OpacityGuard&) = delete; ///< deleted copy constructor
437 OpacityGuard& operator=(const OpacityGuard&) = delete; ///< deleted copy assignment
438 OpacityGuard(OpacityGuard&&) = delete; ///< deleted move constructor
439 OpacityGuard& operator=(OpacityGuard&&) = delete; ///< deleted move assignment
440 };
441 };
