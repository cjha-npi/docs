13 bool ensureParentDirExists(const wchar_t* filePath)
14 {
15 if(filePath == nullptr || filePath[0] == L'\0')
16 return false;
17
18 const std::wstring path(filePath);
19
20 const size_t pos = path.find_last_of(L"\\/");
21 if(pos == std::wstring::npos)
22 return true; // No parent directory component (relative filename only).
23
24 // Parent directory candidate (no trailing slash)
25 std::wstring dir = path.substr(0, pos);
26 if(dir.empty())
27 return false;
28
29 // Handle "C:\file.log" -> dir == "C:" (not a directory path)
30 if(dir.size() == 2 &&
31 ((dir[0] >= L'A' && dir[0] <= L'Z') || (dir[0] >= L'a' && dir[0] <= L'z')) &&
32 dir[1] == L':') {
33 // Drive root is expected to exist; don't try to create it.
34 return true;
35 }
36
37 // Handle UNC root/share: \\server\share\file.log
38 // Do not attempt to create \\server\share (the share itself).
39 if(dir.size() >= 2 && dir[0] == L'\\' && dir[1] == L'\\') {
40 const size_t s1 = dir.find_first_of(L"\\/", 2);
41 if(s1 == std::wstring::npos)
42 return true; // "\\server" only
43
44 const size_t s2 = dir.find_first_of(L"\\/", s1 + 1);
45 if(s2 == std::wstring::npos)
46 return true; // "\\server\\share" (share root)
47 }
48
49 const int rc = ::SHCreateDirectoryExW(nullptr, dir.c_str(), nullptr);
50 return (rc == ERROR_SUCCESS || rc == ERROR_ALREADY_EXISTS || rc == ERROR_FILE_EXISTS);
51 }
